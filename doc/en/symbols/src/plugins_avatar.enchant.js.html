 46 enchant.AvatarCharacter = enchant.Class.create(enchant.Sprite, {
 47     /**
 48      * Constructor of AvatarCharacter
 49      * @param width
 50      * @param height
 51      */
 52 	initialize:function(width,height){
 53 		enchant.Sprite.call(this,width,height);
 54 		this.right();
 55 
 56         /**
 57          * Name of animation pattern;
 58          * @type {String}
 59          */
 60 		this.action = "stop";
 61 		
 62         /**
 63          * Array of animation patterns
 64          * @type {Object}
 65          */
 66 		this.animPattern = { "stop": [ 0]};
 67 
 68         /**
 69          * Frame number of animation
 70          * @type {Object}
 71          */
 72 		this.animFrame=0;
 73 		this.addEventListener('enterframe',function(){
 74 			if((~~(this.age) & 0x03)!=0)return;
 75 			if(this.action){
 76 				var animPattern = this.animPattern[this.action];
 77 				this.frame = animPattern[this.animFrame];
 78 				this.animFrame++;
 79 				if(animPattern[this.animFrame]==-1){
 80 					this.animFrame=0;
 81 					this.action="stop";
 82 				}
 83 				if(animPattern[this.animFrame]==-2){
 84 					this.parentNode.removeChild(this);
 85 				}
 86 				if(this.animFrame>=animPattern.length)
 87 					this.animFrame=0;
 88 			}
 89 		});
 90 	},
 91 	
 92 	/**
 93 	 * Flip to left
 94 	 */
 95 	left:function(){
 96 		this.scaleX=1;
 97 	},
 98 	/**
 99 	 * Flip to right
100 	 */
101 	right:function(){
102 		this.scaleX=-1;
103 	}
104 });

112 enchant.AvatarMonster = enchant.Class.create(enchant.AvatarCharacter, {
113 	/**
114 	 * AvatarMonster
115 	 * Manage a monter animations
116 	 * @param {image} Image of monster
117 	 * @extends enchant.AvatarCharacter
118 	 */
119 	initialize:function(image){
120 		var w = ~~(image.width/4);
121 		var h = w;
122 		enchant.AvatarCharacter.call(this,w,h);
123 		this.image = image;
124 		this.animPattern = { "stop": [ 4,4,4,3,3,3],
125 										"walk":[ 2,3,4,3],
126 										"appear": [ 0,1,7,6,5,4,2,3,-1], 
127 										"disappear": [ 3,2,4,5,6,7,1,0,-2], 
128 										"attack": [ 5,4,6,6,6,-1],
129 									};
130 		this.action="attack";
131 		this.left();
132 	}
133 });

135 /**
136  * AvatarBG
137  * @scope enchant.AvatarBG.prototype
138  */
139 enchant.AvatarBG = enchant.Class.create(enchant.Group, {
140 	/**
141 	 * A class of infinite scrolling background.
142 	 * @param {mode} 0 to 3 
143 	 * @extends enchant.Group
144 	 */
145 	initialize:function(mode){
146 		enchant.Group.call(this);
147 
148 		this.veryfarbg = new Sprite(320,51);
149 		this.veryfarbg.y=0;
150 		this.veryfarbg.image = game.assets["avatarBg2.png"];
151 		this.veryfarbg.frame=mode;
152 		this.addChild(this.veryfarbg);
153 		this.farbgs = [];
154 		for(i=0;i<3;i++){
155 			var bg = new Sprite(320,32);
156 			bg.image = game.assets["avatarBg3.png"];
157 			bg.frame=mode;
158 			bg.x=i*320-320;bg.y=20;
159 			this.farbgs[this.farbgs.length]=bg;
160 			this.addChild(bg);
161 		}
162 		
163 		this.tiles = [];
164 		for(i=0;i<14;i++){
165 			var tile = new Sprite(32,128);
166 			tile.image = game.assets["avatarBg1.png"];
167 			tile.frame=mode;
168 			tile.x=i*31-48;tile.y=48;
169 			this.tiles[this.tiles.length]=tile;
170 			this.addChild(tile);
171 		}
172 	},
173 	/**
174 	 * horizontal scrolling.
175 	  * @param {int} x Offset of x coordinate.
176 	  */
177 	scroll:function(x){
178 		var dx = ~~(x/2)%320,ddx=~~(x)%32*2;
179 		for(i=0;i<14;i++){
180 			this.tiles[i].x=i*31-ddx-48;
181 		}
182 		for(i=0;i<3;i++){
183 			this.farbgs[i].x=i*320-dx-320;
184 		}
185 	}
186 });

188 /**
189  * Avatar
190  * @scope enchant.Avatar.prototype
191  */
192 enchant.Avatar = enchant.Class.create(enchant.AvatarCharacter, {
193 	/**
194 	 * @param {int}code  Avatar code
195 	 * @extends enchant.AvatarCharacter
196 	 */
197 	initialize:function(code){
198 		enchant.AvatarCharacter.call(this,64,64);
199 		if(code){
200 			this.setCode(code);
201 		}else{
202 			this.gender = 1;
203 			this.hairstyle = 1;
204 			this.haircolor = 0;
205 			this.weapon = 2545;
206 			this.armor = 2135;
207 			this.headpiece = 0;
208 			this.loadImage();
209 		}
210 		this.animPattern = { "stop": [ 0],
211 										"run": [ 1,2,3,2], 
212 										"attack": [ 0,2,9,10,11,5,0,0,0,-1], 
213 										"special": [ 0,6,5,13,14,15,0,0,-1], 
214 										"damage": [ 7,7,7,7,0,0,0,-1], 
215 										"dead": [8], 
216 										"demo":[ 1,2,3,2,1,2,3,2,1,2,3,2,1,2,3,2,1,2,3,2,1,2,3,2,0,0,0,0,2,9,10,11,5,0,0,0,
217 										1,2,3,2,1,2,3,2,1,2,3,2,1,2,3,2,1,2,3,2, 0,6,5,13,14,15,0,0]
218 										};
219 
220 
221 	},
222 	/**
223 	 * Reflesh avatar animation image
224 	 */
225 	loadImage:function(){
226 		var ___EnchantAvatarServerURL  ="http://9leap.net/meruru/draw/draw.php";
227 		this.opt="gender="+this.gender+"&job=0&hairstyle="+this.hairstyle+"&haircolor="+this.haircolor+"&weapon="+this.weapon+"&armor="+this.armor+"&headpiece="+this.headpiece+"&invisible=0&x=0&y=0&w=4&h=4&dummy=.gif";
228 		this.src = ___EnchantAvatarServerURL+"?"+this.opt;
229 		(function(that){
230 			var game = enchant.Game.instance;
231 			game.load(that.src,function(){
232 			that.image = game.assets[that.src];
233 		});})(this);	
234 	},
235 	/**
236 	 * Get avatar code from actual object
237 	 * @return {String} code;
238 	 */
239 	getCode:function(){
240 		return this.gender+":"+this.hairstyle+":"+this.haircolor+":"+
241 					this.weapon+":"+this.armor+":"+this.headpiece;
242 	},
243 	/**
244 	 * Set avatar code and reflesh avatar image.
245 	 * @param {String} code;
246 	 */
247 	setCode:function(code){
248 		data = code.split(":");
249 		this.gender = data[0];
250 		this.hairstyle = data[1];
251 		this.haircolor = data[2];
252 		this.weapon = data[3];
253 		this.armor = data[4];
254 		this.headpiece = data[5];
255 		this.loadImage();
256 	}
257 });
