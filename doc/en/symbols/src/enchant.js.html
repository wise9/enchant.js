 70 /**
 71  * Export library classes globally.
 72  *
 73  * When no arguments are delivered, all classes defined in enchant.js as well as all classes defined in
 74  * plugins will be exported. When more than one argument is delivered, by default only classes defined
 75  * in enchant.js will be exported. When you wish to export plugin classes you must explicitly deliver  *  * plugin identifiers as arguments.
 76  *
 77  * @example
 78  *   enchant();     // All classes will be exported.
 79  *   enchant('');   // Only classes in enchant.js will be exported.
 80  *   enchant('ui'); // enchant.js classes and ui.enchant.js classes will be exported.
 81  *
 82  * @param {...String} [modules] Export module. Multiple designations possible.
 83  */
 84 var enchant = function(modules) {
 85     if (modules != null) {
 86         if (!(modules instanceof Array)) {
 87             modules = Array.prototype.slice.call(arguments);
 88         }
 89         modules = modules.filter(function(module) {
 90             return [module].join();
 91         });
 92     }
 93 
 94     (function include(module, prefix) {
 95         var submodules = [];
 96         for (var prop in module) if (module.hasOwnProperty(prop)) {
 97             if (typeof module[prop] == 'function') {
 98                 window[prop] = module[prop];
 99             } else if (typeof module[prop] == 'object' && Object.getPrototypeOf(module[prop]) == Object.prototype) {
100                 if (modules == null) {
101                     submodules.push(prop);
102                 } else {
103                     i = modules.indexOf(prefix + prop);
104                     if (i != -1) {
105                         submodules.push(prop);
106                         modules.splice(i, 1);
107                     }
108                 }
109             }
110         }
111         for (var i = 0, len = submodules.length; i < len; i++) {
112             include(module[submodules[i]], prefix + submodules[i] + '.');
113         }
114     })(enchant, '');
115 
116     if (modules != null && modules.length) {
117         throw new Error('Cannot load module: ' + modules.join(', '));
118     }
119 };

180 /**
181  * Create class.
182  *
183  * When making classes that succeed other classes, the previous class is used as a base with
184  * constructor as default. In order to override the constructor, it is necessary to explicitly
185  * call up the previous constructor to use it.
186  *
187  * @example
188  *   var Ball = Class.create({ // Creates independent class.
189  *       initialize: function(radius) { ... }, // Method definition.
190  *       fall: function() { ... }
191  *   });
192  *
193  *   var Ball = Class.create(Sprite);  // Creates a class succeeding "Sprite"
194  *   var Ball = Class.create(Sprite, { // Creates a class succeeding "Sprite"
195  *       initialize: function(radius) { // Overwrites constructor
196  *          Sprite.call(this, radius*2, radius*2); // Applies previous constructor.
197  *          this.image = game.assets['ball.gif'];
198  *       }
199  *   });
200  *
201  * @param {Function} [superclass] Preceding class.
202  * @param {*} [definition] Class definition.
203  * @static
204  */
205 enchant.Class.create = function(superclass, definition) {
206     if (arguments.length == 0) {
207         return enchant.Class.create(Object, definition);
208     } else if (arguments.length == 1 && typeof arguments[0] != 'function') {
209         return enchant.Class.create(Object, arguments[0]);
210     }
211 
212     for (var prop in definition) if (definition.hasOwnProperty(prop)) {
213         if (typeof definition[prop] == 'object' && Object.getPrototypeOf(definition[prop]) == Object.prototype) {
214             if (!('enumerable' in definition[prop])) definition[prop].enumerable = true;
215         } else {
216             definition[prop] = { value: definition[prop], enumerable: true, writable: true };
217         }
218     }
219     var constructor = function() {
220         if (this instanceof constructor) {
221             constructor.prototype.initialize.apply(this, arguments);
222         } else {
223             return new constructor();
224         }
225     };
226     constructor.prototype = Object.create(superclass.prototype, definition);
227     constructor.prototype.constructor = constructor;
228     if (constructor.prototype.initialize == null) {
229         constructor.prototype.initialize = function() {
230             superclass.apply(this, arguments);
231         };
232     }
233 
234     return constructor;
235 };

518         var listeners = this._listeners[type];
519         if (listeners == null) {
520             this._listeners[type] = [listener];
521         } else if (listeners.indexOf(listener) == -1) {
522             listeners.unshift(listener);

524         }

530      */
531     removeEventListener: function(type, listener) {
532         var listeners = this._listeners[type];
533         if (listeners != null) {
534             var i = listeners.indexOf(listener);
535             if (i != -1) {
536                 listeners.splice(i, 1);
537             }
538         }
539     },
540     /**
541      * Clear EventListener.
542      * @param {String} type Event type.
543      */
544     clearEventListener: function(type) {
545         if(type != null){
546             delete this._listeners[type];
547         }else{
548             this._listeners = {};
549         }
550     },
551     /**
552      * Issue event.

555     dispatchEvent: function(e) {

562             listeners = listeners.slice();
563             for (var i = 0, len = listeners.length; i < len; i++) {
564                 listeners[i].call(this, e);
565             }
566         }
