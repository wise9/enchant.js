 30 /**
 31  * アクションクラス
 32  * 一つ一つのアクションを管理するオブジェクト
 33  * enchant.animation.Timeline に登録して実行する
 34  * @scope enchant.animation.Action.prototype
 35  */
 36 enchant.animation.Action = enchant.Class.create({
 37     /**
 38      * アクションクラスのコンストラクタ。
 39      * 第二引数で親となるタイムラインを指定する。
 40      * 第三引数にオブジェクトを渡すと、パラメータを上書きできる。
 41      * アクションを管理するオブジェクト(タイムラインオブジェクト) と組み合わせて利用する。
 42      * @param timeline
 43      * @param args
 44      */
 45     initialize: function(timeline, args){
 46         /**
 47          * 親となるタイムライン
 48          * @type {enchant.animation.Timeline}
 49          */
 50         this.timeline = timeline;
 51 
 52         /**
 53          * アクションのなか操作するノード
 54          * @type {enchant.Node}
 55          */
 56         this.node = timeline.node;
 57 
 58         /**
 59          * 現在の経過フレーム数
 60          * time: 1,
 61          * @type {Number}
 62          */
 63         this.frame = 0;
 64 
 65         /**
 66          * アクションが持続フレーム数
 67          * アクションを実行したのち、frame++ > time ならばタイムラインの次のアクションに遷移する
 68          * @type {Number}
 69          */
 70         this.time = 0;
 71 
 72         /**
 73          * アクションの実体となる関数
 74          * tick(); が実行されるごとに一度だけ実行される。
 75          * 実行された回数は frame プロパティに記録されている
 76          * @type {Function}
 77          */
 78         this.func = null;
 79 
 80         /**
 81          * アクションが終了する前に実行される
 82          * @type {Function}
 83          */
 84         this.callback = null;
 85 
 86         /**
 87          * アクションがすでに一度でも実行されているかどうか
 88          * @type {Boolean}
 89          */
 90         this.loaded = false;
 91 
 92         /**
 93          * アクションの初回実行時に実行される関数
 94          * @type {Function}
 95          */
 96         this.onload = null;
 97         if(args){
 98             for(var i in args){
 99                 this[i] = args[i];
100             }
101         }
102     },
103     /**
104      * 1フレーム経過するごとに実行する関数
105      * @return {Boolean} [result] アクションが次のフレームでも持続するかどうか
106      */
107     tick: function(){
108         if(!this.loaded){
109             if(this.onload)this.onload.call(this);
110             this.loaded = true;
111         }
112         if(this.func){
113             this.func();
114         }
115         this.frame ++;
116         if(this.time < this.frame){
117             if(this.callback){
118                 this.callback();
119                 return false;
120             }
121         }else{
122             return true;
123         }
124     },
125     /**
126      * 実行されていない状態に戻す
127      */
128     reset: function(){
129         this.frame = 0;
130         this.loaded = false;
131     }
132 });

135 /**
136  * @scope enchant.animation.Tween.prototype
137  */
138 enchant.animation.Tween = enchant.Class.create(enchant.animation.Action, {
139     /**
140      * トゥイーンクラス。
141      * アクションクラスを継承しており、座標や回転・大きさ・透明度などを、時間に従って変更するために使う。
142      * イージング関数を指定することで、様々なイージングを実現できる。
143      * コンストラクタの第二引数で、トゥイーンの詳細を指定する。
144      *
145      * @example
146      * var tween = new Tween(sprite, {
147      *     x: 100,
148      *     y: 0,
149      *     easing: enchant.Easing.QUAD_EASEINOUT
150      * });
151      *
152      * @param {enchant.Node} [node] 親となるノード
153      * @param {*} [args] パラメータ
154      * @constructs
155      * @extends enchant.animation.Action
156      */
157     initialize: function(node, args){
158         this._origin = {};
159         this._target = {};
160         enchant.animation.Action.call(this, node, args);
161 
162         if(typeof this.easing === "undefined"){
163             this.easing = function(t, b, c, d) {
164                 return c*t/d + b;
165             };
166         }
167 
168         if(this.onload == null){
169             this.onload = function(){
170                 // トゥイーンの対象とならないプロパティ
171                 var excepted = ["frame", "time", "callback", "onload"];
172                 for(var i in args){
173                     if(excepted.indexOf(i) == -1){
174                         this._origin[i] = node[i];
175                         this._target[i] = args[i];
176                     }
177                 }
178             }
179         }
180 
181         this.node = node;
182         
183         /**
184          * トゥイーンで毎フレーム実行される関数
185          */
186         this.func = function(){
187             for(var i in this._target){
188                 if(typeof this.node[i] === "undefined")continue;
189                 var ratio = this.easing(this.frame, 0, 1, this.time);
190                 var val = this._target[i] * ratio + this._origin[i] * (1 - ratio);
191                 if(i === "x" || i === "y"){
192                     this.node[i] = Math.round(val);
193                 }else{
194                     this.node[i] = val;
195                 }
196             }
197         }
198     }
199 });

201 /**
202  * タイムラインクラス。
203  * ノードに関するアクションの実行を管理するオブジェクト。
204  * タイムラインに追加されたアクションは順に実行され、すべて実行し終わると待機状態となる。
205  *
206  * タイムライン自身に様々なアクション・トゥイーンを追加するメソッドが用意されており、
207  * メソッドチェーンによりアクションを次々に追加することも可能。
208  * @example
209  * var bear = new AnimationSprite(32, 32);
210  * bear.animation.delay(30).moveTo(0, 320, 60);
211  */
212 enchant.animation.Timeline = enchant.Class.create(
213     {
214     /**
215      * @construct
216      * @param {enchant.Node} [node] アクションの対象となるノード (スプライトなど)
217      */
218     initialize: function(node){
219         this.queue = [];
220         this.node = node;
221         this.paused = false;
222         this.looped = false;
223     },
224     /**
225      * 1フレームごとに実行するメソッド。
226      * 親ノードの、enterframeイベントに対するリスナとして登録される。
227      * queue プロパティの最初のアクションを実行する。
228      * 返値が false ならば、次のアクションを実行する。
229      */
230     tick: function(){
231         if(this.paused) return;
232         if(this.queue.length == 0){
233             // do nothing
234         }else{
235             if(this.queue[0].tick()){
236                 // ticked
237             }else{
238                 // end
239                 if(this.looped){
240                     // 再度キューに追加する
241                     var action = this.queue.shift();
242                     action.reset();
243                     this.queue.push(action);
244                 }else{
245                     this.queue.shift();
246                 }
247                 this.tick();
248             }
249         }
250     },
251     /**
252      * トゥイーンを生成し、追加する。
253      * 引数で生成するトゥイーンの詳細を指定できる。
254      * (すでに生成したトゥイーンを実行待ちキューに追加するには、
255      *  timeline.queue.push(tween) を用いる。)
256      * @param {Number} args
257      * @time {Number} フレーム数 optional
258      */
259     pushTween: function(args, time){
260         var tween = new enchant.animation.Tween(this.node, args);
261         if(time !== undefined)tween.time = time;
262         this.queue.push(tween);
263         return this;
264     },
265     /**
266      * アクションを生成し、追加する。
267      * 引数で生成するアクションの詳細を指定できる。
268      * (すでに生成したアクションを実行待ちキューに追加するには、
269      *  timeline.queue.push(tween) を用いる。)
270      * @param {Number} args
271      * @time {Number} フレーム数 optional
272      */
273     pushAction: function(args, time){
274         var action = new enchant.animation.Action(this, args)
275         if(time !== undefined)tween.time = time;
276         this.queue.push(action);
277         return this;
278     },
279     /**
280      * 不透明度を遷移させるトゥイーンを生成し、キューに追加するメソッド
281      * @param {Number} opacity 目標値
282      * @param {Number} time フレーム数
283      * @param {Function} easing トゥイーン関数
284      */
285     fadeTo: function(opacity, time, easing){
286         if(arguments.length < 2)time = enchant.Game.instance.fps;
287         if(arguments.length < 3)easing = enchant.Easing.LINEAR;
288         this.pushTween({
289             time: time,
290             opacity: opacity,
291             easing: easing
292         });
293         return this;
294     },
295     /**
296      * 不透明度を現在の値から1に遷移させるトゥイーンを生成し、キューに追加するメソッド
297      * フェードインが実現できる。
298      * シーンからの removeChild は実行されないので注意
299      * @param {Number} time フレーム数
300      * @param {Function} easing トゥイーン関数
301      */
302     fadeIn: function(time, easing){
303         if(time === undefined)time = enchant.Game.instance.fps;
304         if(easing === undefined)easing = enchant.Easing.LINEAR;
305         this.fadeTo(1, time, easing);
306         return this;
307     },
308     /**
309      * 不透明度を現在の値から1に遷移させるトゥイーンを生成し、キューに追加するメソッド
310      * フェードインが実現できる。
311      * シーンへの addChild は実行されないので注意
312      * @param {Number} time フレーム数
313      * @param {Function} easing トゥイーン関数
314      */
315     fadeOut: function(time, easing){
316         if(time === undefined)time = enchant.Game.instance.fps;
317         if(easing === undefined)easing = enchant.Easing.LINEAR;
318         this.fadeTo(0, time, easing);
319         return this;
320     },
321     /**
322      * Entity を画面から消すアクションを追加。
323      * removeChild しているわけではなく、透明になるだけなので (opacity = 0)、
324      * クリックイベントなどは受け取ることができる。
325      */
326     hide: function(){
327         this.pushAction({
328             time: 0,
329             callback: function(){
330                 this.node.opacity = 0;
331             }
332         });
333         return this;
334     },
335     /**
336      * hide() や fadeOut() などで画面から消した
337      * Entityを再度出現させるアクションを追加 (opacity = 1)
338      */
339     show: function(){
340         this.pushAction({
341             time: 0,
342             callback: function(){
343                 this.opacity = 0;
344             }
345         })
346     },
347     /**
348      * 縮尺を変化させるトゥイーンを生成し、キューに追加するメソッド
349      * @param {Number} scale 目標値
350      * @param {Number} time フレーム数
351      * @param {Function} easing トゥイーン関数
352      */
353     scaleTo: function(scale, time, easing){
354         if(arguments.length < 2)time = enchant.Game.instance.fps;
355         if(arguments.length < 3)easing = enchant.Easing.LINEAR;
356         this.pushTween({
357             time: time,
358             scaleX: scale,
359             scaleY: scale,
360             easing: easing
361         });
362         return this;
363     },
364     /**
365      * 縮尺を変化させるトゥイーンを生成し、キューに追加するメソッド
366      * @param {Number} scale 目標値
367      * @param {Number} time フレーム数
368      * @param {Function} easing トゥイーン関数
369      */
370     scaleBy: function(scale, time, easing){
371         if(time === undefined)time = enchant.Game.instance.fps;
372         if(easing === undefined)easing = enchant.Easing.LINEAR;
373         this.pushTween({
374             time: time,
375             onload: function(){
376                 this._origin.scaleX = this.node.scaleX;
377                 this._target.scaleX = this.node.scaleX * scale;
378                 this._origin.scaleY = this.node.scaleY;
379                 this._target.scaleY = this.node.scaleY * scale;
380             },
381             scale: scale,
382             easing: easing
383         });
384         return this;
385     },
386     /**
387      * 回転するトゥイーンを生成し、キューに追加するメソッド
388      * rotation プロパティの値を、現在の値から指定した値に変化させる
389      * @param {Number} deg 回転角度 (度数法)
390      * @param {Number} time フレーム数
391      * @param {Function} easing トゥイーン関数
392      */
393     rotateTo: function(deg, time, easing){
394         if(time === undefined)time = enchant.Game.instance.fps;
395         if(easing === undefined)easing = enchant.Easing.LINEAR;
396         this.pushTween({
397             time: time,
398             rotation: deg,
399             easing: easing
400         });
401         return this;
402     },
403     /**
404      * 回転するトゥイーンを生成し、キューに追加するメソッド
405      * rotation プロパティの値を、現在の値から指定した値だけ変化させる
406      * @param {Number} deg 回転角度 (度数法)
407      * @param {Number} time フレーム数
408      * @param {Function} easing トゥイーン関数
409      */
410     rotateBy: function(deg, time, easing){
411         if(time === undefined)time = enchant.Game.instance.fps;
412         if(easing === undefined)easing = enchant.Easing.LINEAR;
413         this.pushTween({
414             time: time,
415             onload: function(){
416                 this._origin.rotation = this.node.rotation;
417                 this._target.rotation = this.node.rotation + deg;
418             },
419             rotation: deg,
420             easing: easing
421         });
422         return this;
423     },
424     /**
425      * 移動するトゥイーンを生成し、キューに追加するメソッド
426      * x, y座標を、現在の値から指定した値に変化させる
427      * @param {Number} x 移動先のx座標
428      * @param {Number} y 移動先のy座標
429      * @param {Number} time 角度
430      * @param {Function} easing トゥイーン関数
431      */
432     moveTo: function(x, y, time, easing){
433         if(time === undefined)time = enchant.Game.instance.fps;
434         if(easing === undefined)easing = enchant.Easing.LINEAR;
435         this.pushTween({
436             time: time,
437             x: x,
438             y: y,
439             easing: easing
440         });
441         return this;
442     },
443     /**
444      * 移動するトゥイーンを生成し、キューに追加するメソッド
445      * x, y座標を、現在の値から指定した値だけ変化させる
446      * @param {Number} x x座標の移動量
447      * @param {Number} y y座標の移動量
448      * @param {Number} time 角度
449      * @param {Function} easing トゥイーン関数
450      */
451     moveBy: function(x, y, time, easing){
452         if(time === undefined)time = enchant.Game.instance.fps;
453         if(easing === undefined)easing = enchant.Easing.LINEAR;
454         this.pushTween({
455             time: time,
456             onload: function(){
457                 this._origin.x = this.node.x;
458                 this._origin.y = this.node.y;
459                 this._target.x = this.node.x + x;
460                 this._target.y = this.node.y + y;
461             },
462             easing: easing
463         });
464         return this;
465     },
466     /**
467      * 何もせず、指定されたフレーム数だけ待つアクションをキューに追加する
468      * @param time
469      */
470     delay: function(time){
471         this.pushAction({
472             time: time
473         });
474         return this;
475     },
476     /**
477      *
478      * 指定した関数を1度だけ実行するアクションをキューに追加する
479      * delayと組み合わせて、一定フレーム後に実行する関数を指定することもできる
480      * @param func 実行する関数
481      */
482     then: function(func){
483         this.pushAction({
484             time: 0,
485             callback: func
486         });
487         return this;
488     },
489     /**
490      * 指定した関数を、返り値が true になるまで毎フレーム実行するアクションをキューに追加する
491      * @param func 実行する関数
492      */
493     after: function(func){
494         this.pushAction({
495             tick: function(){
496                 return !func();
497             }
498         });
499         return this;
500     },
501     /**
502      * 指定した関数を、返り値が true になるまで毎フレーム実行するアクションをキューに追加する
503      * @param func 実行する関数
504      */
505     repeat: function(func, time, span){
506         if(span === undefined)span = 1;
507         this.pushAction({
508             func: function(){
509                 if(this.frame % span == 0){
510                     func();
511                 }
512             },
513             time: time
514         });
515         return this;
516     },
517     loop: function(actions){
518         return this;
519     },
520     /**
521      * x,y 座標をディレイなしで変更するアクションをキューに追加する
522      * @param x
523      * @param y
524      */
525     jumpTo: function(x, y){
526         var node = this.node;
527         return this.then(function(){
528             node.x = x;
529             node.y = y;
530             });
531     },
532     /**
533      * x,y 座標をディレイなしで変更するアクションをキューに追加する
534      * @param x
535      * @param y
536      */
537     jumpBy: function(x, y){
538         var node = this.node;
539         return this.then(function(){
540             node.x = node.x + x;
541             node.y = node.y + y;
542             });
543     },
544     /**
545      * アニメーションキューの実行をストップする
546      */
547     pause: function(){
548         this.paused = true;
549         return this;
550     },
551     /**
552      * アニメーションキューの実行を再開する
553      */
554     start: function(){
555         this.paused = false;
556         return this;
557     },
558     /**
559      * アニメーションキューを空にする
560      */
561     clear: function(){
562         this.queue = [];
563         this.paused = false;
564         return this;
565     }
566 })

653 /**
654  * イージング関数ライブラリ
655  * ActionScript で広く使われている
656  * Robert Penner による Easing Equations を JavaScript に移植した。
657  * @scope enchant.Easing
658  */
659 enchant.Easing = {
660     LINEAR: function(t, b, c, d) {
661 		return c*t/d + b;
662     },
663 	// quad
664 	QUAD_EASEIN: function(t, b, c, d) {
665 		return c*(t/=d)*t + b;
666 	},
667 	QUAD_EASEOUT: function(t, b, c, d) {
668 		return -c *(t/=d)*(t-2) + b;
669 	},
670 	QUAD_EASEINOUT: function(t, b, c, d) {
671 		if ((t/=d/2) < 1) return c/2*t*t + b;
672 		return -c/2 * ((--t)*(t-2) - 1) + b;
673 	},
674 	// qubic
675 	QUBIC_EASEIN: function(t, b, c, d) {
676 		return c*(t/=d)*t*t + b;
677 	},
678 	QUBIC_EASEOUT: function(t, b, c, d) {
679 		return c*((t=t/d-1)*t*t + 1) + b;
680 	},
681 	QUBIC_EASEINOUT: function(t, b, c, d) {
682 		if ((t/=d/2) < 1) return c/2*t*t*t + b;
683 		return c/2*((t-=2)*t*t + 2) + b;
684 	},
685 	// quart
686 	QUART_EASEIN: function(t, b, c, d) {
687 		return c*(t/=d)*t*t*t + b;
688 	},
689 	QUART_EASEOUT: function(t, b, c, d) {
690 		return -c * ((t=t/d-1)*t*t*t - 1) + b;
691 	},
692 	QUART_EASEINOUT: function(t, b, c, d) {
693 		if ((t/=d/2) < 1) return c/2*t*t*t*t + b;
694 		return -c/2 * ((t-=2)*t*t*t - 2) + b;
695 	},
696 	// quint
697 	QUINT_EASEIN: function(t, b, c, d) {
698 		return c*(t/=d)*t*t*t*t + b;
699 	},
700 	QUINT_EASEOUT: function(t, b, c, d) {
701 		return c*((t=t/d-1)*t*t*t*t + 1) + b;
702 	},
703 	QUINT_EASEINOUT: function(t, b, c, d) {
704 		if ((t/=d/2) < 1) return c/2*t*t*t*t*t + b;
705 		return c/2*((t-=2)*t*t*t*t + 2) + b;
706 	},
707     //sin
708 	SIN_EASEIN: function(t, b, c, d) {
709 		return -c * Math.cos(t/d * (Math.PI/2)) + c + b;
710 	},
711 	SIN_EASEOUT: function(t, b, c, d) {
712 		return c * Math.sin(t/d * (Math.PI/2)) + b;
713 	},
714 	SIN_EASEINOUT: function(t, b, c, d) {
715 		return -c/2 * (Math.cos(Math.PI*t/d) - 1) + b;
716 	},
717 	// circ
718 	CIRC_EASEIN: function(t, b, c, d) {
719 		return -c * (Math.sqrt(1 - (t/=d)*t) - 1) + b;
720 	},
721 	CIRC_EASEOUT: function(t, b, c, d) {
722 		return c * Math.sqrt(1 - (t=t/d-1)*t) + b;
723 	},
724 	CIRC_EASEINOUT: function(t, b, c, d) {
725 		if ((t/=d/2) < 1) return -c/2 * (Math.sqrt(1 - t*t) - 1) + b;
726 		return c/2 * (Math.sqrt(1 - (t-=2)*t) + 1) + b;
727 	},
728 	// elastic
729     ELASTIC_EASEIN: function (t, b, c, d, a, p) {
730 		if (t==0) return b;  if ((t/=d)==1) return b+c;  if (!p) p=d*.3;
731 		if (!a || a < Math.abs(c)) { a=c; var s=p/4; }
732 		else var s = p/(2*Math.PI) * Math.asin (c/a);
733 		return -(a*Math.pow(2,10*(t-=1)) * Math.sin( (t*d-s)*(2*Math.PI)/p )) + b;
734 	},
735 	ELASTIC_EASEOUT: function (t, b, c, d, a, p) {
736 		if (t==0) return b;  if ((t/=d)==1) return b+c;  if (!p) p=d*.3;
737 		if (!a || a < Math.abs(c)) { a=c; var s=p/4; }
738 		else var s = p/(2*Math.PI) * Math.asin (c/a);
739 		return (a*Math.pow(2,-10*t) * Math.sin( (t*d-s)*(2*Math.PI)/p ) + c + b);
740 	},
741 	ELASTIC_EASEINOUT: function (t, b, c, d, a, p) {
742 		if (t==0) return b;  if ((t/=d/2)==2) return b+c;  if (!p) p=d*(.3*1.5);
743 		if (!a || a < Math.abs(c)) { a=c; var s=p/4; }
744 		else var s = p/(2*Math.PI) * Math.asin (c/a);
745 		if (t < 1) return -.5*(a*Math.pow(2,10*(t-=1)) * Math.sin( (t*d-s)*(2*Math.PI)/p )) + b;
746 		return a*Math.pow(2,-10*(t-=1)) * Math.sin( (t*d-s)*(2*Math.PI)/p )*.5 + c + b;
747 	},
748     // bounce
749 	BOUNCE_EASEOUT: function(t, b, c, d) {
750 		if ((t/=d) < (1/2.75)) {
751 			return c*(7.5625*t*t) + b;
752 		} else if (t < (2/2.75)) {
753 			return c*(7.5625*(t-=(1.5/2.75))*t + .75) + b;
754 		} else if (t < (2.5/2.75)) {
755 			return c*(7.5625*(t-=(2.25/2.75))*t + .9375) + b;
756 		} else {
757 			return c*(7.5625*(t-=(2.625/2.75))*t + .984375) + b;
758 		}
759 	},
760 	BOUNCE_EASEIN: function(t, b, c, d) {
761 		return c - enchant.Easing.BOUNCE_EASEOUT(d-t, 0, c, d) + b;
762 	},
763 	BOUNCE_EASEINOUT: function(t, b, c, d) {
764 		if (t < d/2) return enchant.Easing.BOUNCE_EASEIN(t*2, 0, c, d) * .5 + b;
765 		else return enchant.Easing.BOUNCE_EASEOUT(t*2-d, 0, c, d) * .5 + c*.5 + b;
766 	},
767     // back
768 	BACK_EASEIN: function(t, b, c, d, s) {
769 		if (s == undefined) s = 1.70158;
770 		return c*(t/=d)*t*((s+1)*t - s) + b;
771 	},
772 	BACK_EASEOUT: function(t, b, c, d, s) {
773 		if (s == undefined) s = 1.70158;
774 		return c*((t=t/d-1)*t*((s+1)*t + s) + 1) + b;
775 	},
776 	BACK_EASEINOUT: function(t, b, c, d, s) {
777 		if (s == undefined) s = 1.70158;
778 		if ((t/=d/2) < 1) return c/2*(t*t*(((s*=(1.525))+1)*t - s)) + b;
779 		return c/2*((t-=2)*t*(((s*=(1.525))+1)*t + s) + 2) + b;
780 	},
781 	// expo
782 	EXPO_EASEIN: function(t, b, c, d) {
783 		return (t==0) ? b : c * Math.pow(2, 10 * (t/d - 1)) + b;
784 	},
785 	EXPO_EASEOUT: function(t, b, c, d) {
786 		return (t==d) ? b+c : c * (-Math.pow(2, -10 * t/d) + 1) + b;
787 	},
788 	EXPO_EASEINOUT: function(t, b, c, d) {
789 		if (t==0) return b;
790 		if (t==d) return b+c;
791 		if ((t/=d/2) < 1) return c/2 * Math.pow(2, 10 * (t - 1)) + b;
792 		return c/2 * (-Math.pow(2, -10 * --t) + 2) + b;
793 	}
794 };
