 24 // 画像でフォントを再現したラベル (参考: draw.text.js)
 25 enchant.util.MutableText = enchant.Class.create(enchant.Sprite, {
 26 	initialize: function(posX, posY, width, height) {
 27 		enchant.Sprite.call(this, 0, 0);
 28 		var width = (arguments[2] || enchant.Game.instance.width);
 29 		var height = (arguments[3] || enchant.Game.instance.height);
 30 		this.fontSize = 16;
 31 		this.widthItemNum = 16;
 32 		// font.png の横の文字数
 33 		this.returnLength = width/this.fontSize;
 34 		// 改行を入れる文字数
 35 		this.image = new Surface(width, height);
 36 		this.x = posX;
 37 		this.y = posY;
 38 		this.text = '';
 39 	},
 40 	setText: function(txt) {
 41 		var i, x, y, wNum, charCode, charPos;
 42 		this._text = txt;
 43 		this.width = this.returnLength * this.widthItemNum;
 44 		this.height = this.fontSize * (((this._text.length / this.returnLength)|0)+1);
 45 		this.image.context.clearRect(0, 0, this.width, this.height);
 46 		for(i=0; i<txt.length; i++) {
 47 			charCode = txt.charCodeAt(i);
 48 			if (charCode >= 32 && charCode <= 127) {
 49 				charPos = charCode - 32;
 50 			} else {
 51 				charPos = 0;
 52 			}
 53 			x = charPos % this.widthItemNum;
 54 			y = (charPos / this.widthItemNum)|0;
 55 			this.image.draw(enchant.Game.instance.assets['font.png'], 
 56 				x * this.fontSize, y * this.fontSize, this.fontSize, this.fontSize,
 57 				(i%this.returnLength)*this.fontSize, ((i/this.returnLength)|0)*this.fontSize, this.fontSize, this.fontSize);
 58 		}
 59 	},
 60 	text: {
 61 		get: function() {
 62 			return this._text;
 63 		},
 64 		set: function(txt) {
 65 			this.setText(txt);
 66 		}
 67 	},
 68 	row: {
 69 		get: function() {
 70 			return this.returnLength;
 71 		},
 72 		set: function(row) {
 73 			this.returnLength = row;
 74 			this.text = this.text;
 75 		}
 76 	}
 77 });

 79 // スコアラベル
 80 enchant.util.ScoreLabel = enchant.Class.create(enchant.util.MutableText, {
 81 	initialize: function(x, y) {
 82 		MutableText.call(this, 0, 0);
 83 		switch(arguments.length){
 84 			case 2:
 85 				this.y = y;
 86 			case 1:
 87 				this.x = x;
 88 				break;
 89 			default:
 90 				break;
 91 		}
 92 		this._score = 0;
 93 		this._current = 0;
 94 		this.easing = 2.5;
 95 		this.text = this.label = 'SCORE:';
 96 		this.addEventListener('enterframe', function(){
 97 			if(this.easing == 0){
 98 				this.text = this.label + (this._current = this._score);
 99 			}else{
100 				this._current += Math.ceil((this._score - this._current)/this.easing);
101 				this.text = this.label + this._current;
102 			}
103 		});
104 	},
105 	score: {
106 		get: function() {
107 			return this._score;
108 		},
109 		set: function(newscore){
110 			this._score = newscore;
111 		}
112 	}
113 });

116 enchant.util.TimeLabel = enchant.Class.create(enchant.util.MutableText, {
117 	initialize: function(x, y, counttype) {
118 		MutableText.call(this, 0, 0);
119 		switch(arguments.length){
120 			case 3:
121 			case 2:
122 				this.y = y;
123 			case 1:
124 				this.x = x;
125 				break;
126 			default:
127 				break;
128 		}
129 		this._time = 0;
130 		this._count = 1;// この数を毎フレーム每に足して上げ下げを制御する
131 		if(counttype == 'countdown')this._count = -1;
132 		this.text = this.label = 'TIME:';
133 		this.addEventListener('enterframe', function(){
134 			this._time += this._count;
135 			this.text = this.label + (this._time / enchant.Game.instance.fps).toFixed(2);
136 		});
137 	},
138 	time: {
139 		get: function() {
140 			return Math.floor(this._time / enchant.Game.instance.fps);
141 		},
142 		set: function(newtime){
143 			this._time = newtime * enchant.Game.instance.fps;
144 		}
145 	}

149 enchant.util.LifeLabel = enchant.Class.create(enchant.Group, {
150 	initialize: function(x, y, maxlife) {
151 		Group.call(this);
152 		this.x = x || 0;
153 		this.y = y || 0;
154 		this._maxlife = maxlife || 9;
155 		this._life = 0;
156 		this.label = new MutableText(0, 0, 80, 'LIFE:')
157 		this.addChild(this.label);
158 		this.heart = [];
159 		for(var i=0; i<this._maxlife; i++){
160 			this.heart[i] = new Sprite(16, 16);
161 			this.heart[i].image = enchant.Game.instance.assets['icon0.gif'];
162 			this.heart[i].x = this.label.width + i*16;
163 			this.heart[i].y = -3;
164 			this.heart[i].frame = 11;
165 			this.addChild(this.heart[i]);
166 		}
167 	},
168 	life: {
169 		get: function() {
170 			return this._life;
171 		},
172 		set: function(newlife) {
173 			this._life = newlife;
174 			if(this._maxlife < newlife){
175 				this._life = this._maxlife;
176 			}
177 			for(var i=0; i<this._maxlife; i++){
178 				if(i <= newlife-1){
179 					this.heart[i].visible = true;
180 				}else{
181 					this.heart[i].visible = false;
182 				}
183 			}
184 		}
185 	}
186 });

188 // イージング付きのバー (左右方向のみ) 
189 enchant.Bar = enchant.Class.create(enchant.Sprite, {
190 	initialize: function(x, y) {
191 		Sprite.call(this, 1, 16);
192 		this.image = new Surface(1, 16);// Null用
193 		this.image.context.fillColor = 'RGB(0, 0, 256)';
194 		this.image.context.fillRect(0, 0, 1, 16);
195 		this._direction = 'right';
196 		this._origin = 0;
197 		this._maxvalue = enchant.Game.instance.width;
198 		this._lastvalue = 0;
199 		this.value = 0;
200 		this.easing = 5;
201 		switch(arguments.length){
202 			case 2:
203 				this.y = y;
204 			case 1:
205 				this.x = x;
206 				this._origin = x;
207 			default:
208 				break;
209 		}
210 		this.addEventListener('enterframe', function(){
211 			if(this.value < 0)this.value = 0;
212 			this._lastvalue += (this.value - this._lastvalue)/this.easing;
213 			if(Math.abs(this._lastvalue - this.value) < 1.3)this._lastvalue = this.value;
214 			this.width = (this._lastvalue)|0;
215 			if(this.width > this._maxvalue)this.width = this._maxvalue;
216 			if(this._direction == 'left'){
217 				this._x = this._origin - this.width;
218 			}else{
219 				this._x = this._origin;
220 			}
221 			this._updateCoordinate();
222 		});
223 	},
224 	direction: {
225 		get: function() {
226 			return this._direction;
227 		},
228 		set: function(newdirection) {
229 			if(newdirection != 'right' && newdirection != 'left'){
230 				console.warn(newdirection + ' は未定義の向きです。rightかleftを指定してください.');
231 			}else{
232 				this._direction = newdirection;
233 			}
234 		}
235 	},
236 	x: {
237 		get: function() {
238 			return this._origin;
239 		},
240 		set: function(x) {
241 			this._x = x;
242 			this._origin = x;
243 			this._updateCoordinate();
244 		}
245 	},
246 	maxvalue: {
247 		get: function(){
248 			return this._maxvalue;
249 		},
250 		set: function(val){
251 			this._maxvalue = val;
252 		}
253 	}
254 });

256 /**
257  * マップライクな Group
258  * addChildで Sprite 等を追加すると、自動的に mx, my プロパティが追加され、
259  * VirtualMap内での座標で Sprite を操作できる
260  *
261  * 使い方
262  * //20 x 20 メッシュの縦横320ピクセルの盤を作り、その上に16 x 16の駒を8つ並べる
263  * var board = new VirtualMap(20, 20);
264  * board.width = 320;
265  * board.height = 320;
266  * for(var i=0; i<8; i++){
267  *     var piece = new Sprite(16, 16);
268  *     piece.image = game.assets['icon0.gif'];
269  *     board.addChild(piece);
270  *     piece.mx = i + 3;
271  *     piece.my = 16;
272  * }
273  * game.rootScene.addChild(board);
274  */
275 enchant.util.VirtualMap = enchant.Class.create(enchant.Group, {
276     initialize: function (meshWidth, meshHeight) {
277         enchant.Group.call(this);
278         this.meshWidth = meshWidth || 16;
279         this.meshHeight = meshHeight || 16;
280     },
281     addChild: function (obj) {
282         enchant.Group.prototype.addChild.call(this, obj);
283         this.bind(obj);
284     },
285     insertBefore: function (obj, reference) {
286         enchant.Group.prototype.insertBefore.call(this, obj, reference);
287         this.bind(obj);
288     },
289     bind: function (obj) {
290         Object.defineProperties(obj, {
291             "mx": {
292                 get: function () {
293                     return Math.floor(this.x / this.parentNode.meshWidth);
294                 },
295                 set: function (arg) {
296                     this.x = Math.floor(arg * this.parentNode.meshWidth);
297                 }
298             },
299             "my": {
300                 get: function () {
301                     return Math.floor(this.y / this.parentNode.meshHeight);
302                 },
303                 set: function (arg) {
304                     this.y = Math.floor(arg * this.parentNode.meshWidth);
305                 }
306             }
307         });
308         obj.mx = 0;
309         obj.my = 0;
310     }
311 });
