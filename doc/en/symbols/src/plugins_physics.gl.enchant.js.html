 23          * @scope enchant.gl.physics.World.prototype
 24          */
 25         enchant.gl.physics.World = enchant.Class.create({
 26             /**
 27              * World in which physical calculator is used.
 28              * Here we add a rigid object, and by proceeding forward in time physical calculation will be processed.
 29              * @see enchant.gl.physics.PhyScene3D
 30              * @constructs
 31              */
 32             initialize: function() {
 33                 var g = new Ammo.btVector3(0, -10, 0);
 34                 var collisionConfiguration = new Ammo.btDefaultCollisionConfiguration();
 35                 var dispatcher = new Ammo.btCollisionDispatcher(collisionConfiguration);
 36                 var pairCache = new Ammo.btDbvtBroadphase();
 37                 var constraintSolver = new Ammo.btSequentialImpulseConstraintSolver();
 38 
 39                 this._dynamicsWorld = new Ammo.btDiscreteDynamicsWorld(
 40                     dispatcher, pairCache, constraintSolver, collisionConfiguration);
 41 
 42                 this._dynamicsWorld.setGravity(g);
 43 
 44                 Ammo.destroy(g);
 45             },
 46             /**
 47              * Set gravity of World.
 48              * @param {Number} gx x axis gravity.
 49              * @param {Number} gy y axis gravity.
 50              * @param {Number} gz z axis gravity.
 51              */
 52             setGravity: function(gx, gy, gz) {
 53                 var g = new Ammo.btVector3(gx, gy, gz);
 54                 this._dynamicsWorld.setGravity(g);
 55                 Ammo.destroy(g);
 56             },
 57             /**
 58              * Continue World time.
 59              * When timeStep is bigger than fixedTimeStep, you continue the number of times set in maxSubStep and continue with time.
 60              * @param {Number} Amount of time you wish to proceed. Units are in seconds.
 61              * @param {Number} maxSubSteps Maximum number of repetitions to be added to simulation.
 62              * @param {Number} fixedTimeStep Standard time. The default level is 1/60.
 63              * @return {Number} subStepsNum
 64              */
 65             stepSimulation: function(timeStep, maxSubSteps, fixedTimeStep) {
 66                 return this._dynamicsWorld.stepSimulation(timeStep, maxSubSteps, fixedTimeStep);
 67             },
 68             /**
 69              * Add rigid body to World.
 70              * @param {enchant.gl.physics.Rigid} Rigid Rigid body object to be added.
 71              */
 72             addRigid: function(rigid) {
 73                 this._dynamicsWorld.addRigidBody(rigid.rigidBody);
 74                 rigid.world = this;
 75             },
 76             /**
 77              * Delete rigid body form world.
 78              * @param {enchant.gl.physics.Rigid} Rigid Rigid body to be deleted.
 79              */
 80             removeRigid: function(rigid) {
 81                 this._dynamicsWorld.removeRigidBody(rigid.rigidBody);
 82                 rigid.world = null;
 83             },
 84             /**
 85              * Judges whether or not two rigid objects have collided.
 86              * @param {enchant.gl.physics.Rigid} rigid1 Rigid1 to be judged.
 87              * @param {enchant.gl.physics.Rigid} rigid2 Rigid2 to be judged.
 88              * @return {Boolean} bool Collison presence or absence.
 89              */
 90             contactPairTest: function(rigid1, rigid2) {
 91                 var callback = new Ammo.ConcreteContactResultCallback();
 92                 var result = false;
 93                 Ammo.customizeVTable(callback, [{
 94                     original: Ammo.ConcreteContactResultCallback.prototype.addSingleResult,
 95                     replacement: function(tp, cp, colObj0, partid0, index0, colObj1, partid1, index1) {
 96                         result = true;
 97                     }
 98                 }]);
 99                 this._dynamicsWorld.contactPairTest(rigid1.rigidBody, rigid2.rigidBody, callback);
100                 Ammo.destroy(callback);
101                 return result;
102             }
103         });

105         /**
106          * @scope enchant.gl.physics.Rigid.prototype
107          */
108         enchant.gl.physics.Rigid = enchant.Class.create({
109             /**
110              * Rigid body object.
111              * Add to World and activate.
112              * @param shape Ammo.btCollisionShape object.
113              * @param {Number} mass Mass of rigid object.
114              * @param {Number} linearDamping Rate of decay of speed on straight line speed of rigid object.
115              * @param {Number} angularDamping Rate of decay of angled speed of rigid object.
116              * @see enchant.gl.physics.RigidBox
117              * @see enchant.gl.physics.RigidCube
118              * @see enchant.gl.physics.RigidSphere
119              * @see enchant.gl.physics.RigidCylinder
120              * @see enchant.gl.physics.RigidCapsule
121              * @see enchant.gl.physics.RigidPlane
122              * @see enchant.gl.physics.RigidContainer
123              * @constructs
124              */
125             initialize: function(shape, mass, lDamp, aDamp) {
126                 if (typeof shape == 'undefined') {
127                     shape = new Ammo.btBoxShape(1);
128                 } if (typeof mass == 'undefined') {
129                     mass = 1;
130                 }
131 
132                 var localInertia = new Ammo.btVector3(0, 0, 0);
133                 shape.calculateLocalInertia(mass, localInertia);
134 
135                 var transform = new Ammo.btTransform();
136                 transform.setIdentity();
137 
138                 var motionState = new Ammo.btDefaultMotionState(transform);
139                 var rigidBodyInfo = new Ammo.btRigidBodyConstructionInfo(mass, motionState, shape, localInertia);
140                 rigidBodyInfo.set_m_restitution(0.1);
141                 rigidBodyInfo.set_m_friction(3.0);
142 
143                 if (typeof lDamp != 'undefined') {
144                     rigidBodyInfo.set_m_linearDamping(lDamp);
145                 } if (typeof aDamp != 'undefined') {
146                     rigidBodyInfo.set_m_angularDamping(aDamp);
147                 }
148 
149                 this.shape = shape;
150                 /**
151                  * World that Rigid belongs to
152                  */
153                 this.world = null;
154 
155                 /**
156                  * Ammo rigid body object
157                  */
158                 this.rigidBody = new Ammo.btRigidBody(rigidBodyInfo);
159                 var p = Ammo.getPointer(this.rigidBody);
160                 enchant.gl.physics.Rigid._refs[p] = this;
161 
162                 Ammo.destroy(transform);
163                 Ammo.destroy(localInertia);
164                 Ammo.destroy(rigidBodyInfo);
165 
166                 this._x = 0;
167                 this._y = 0;
168                 this._z = 0;
169                 this._scaleX = 1;
170                 this._scaleY = 1;
171                 this._scaleZ = 1;
172                 this._mass = mass;
173                 this._restitution = 0.3;
174                 this._friction = 0.3;
175             },
176             /**
177              * Expand and contract Rigid.
178              * Expand or contract each axis the designated amount from the current rate of expansion in World.
179              * @param {Number} x Rate of expansion on x axis.
180              * @param {Number} y Rate of expansion on y axis.
181              * @param {Number} z Rate of expansion on z axis.
182              */
183             scale: function(x, y, z) {
184                 this.activate();
185                 this._scaleX *= x;
186                 this._scaleY *= y;
187                 this._scaleZ *= z;
188                 var scale = new Ammo.btVector3(this._scaleX, this._scaleY, this._scaleZ);
189                 this.shape.setLocalScaling(scale);
190                 Ammo.destroy(scale);
191             },
192             _scaleAxis: function(axis, scale) {
193                 axis.toUpperCase();
194                 this['_scale' + axis] = scale;
195                 var scale = new Ammo.btVector3(this._scaleX, this._scaleY, this._scaleZ);
196                 this.shape.setLocalScaling(scale);
197                 Ammo.destroy(scale);
198             },
199             /**
200              * Move Rigid in a parallel direciton.
201              * Move the designated amount in a parallel direction along each axis from the current World position.
202              * @param {Number} x Amount of parallel movement along x axis.
203              * @param {Number} y Amount of parallel movement along y axis.
204              * @param {Number} z Amount of parallel movement along z axis.
205              */
206             translate: function(x, y, z) {
207                 this.activate();
208                 var vec = new Ammo.btVector3(x, y, z);
209                 this.rigidBody.translate(vec);
210                 Ammo.destroy(vec);
211                 this._x += x;
212                 this._y += y;
213                 this._z += z;
214             },
215             _translateAxis: function(axis, n) {
216                 this.activate();
217                 var x = 0;
218                 var y = 0;
219                 var z = 0;
220                 if (axis == 'x') {
221                     x = n - this._x;
222                     this._x = n;
223                 } else if (axis == 'y') {
224                     y = n - this._y;
225                     this._y = n;
226                 } else if (axis == 'z') {
227                     z = n - this._z;
228                     this._z = n;
229                 }
230                 var vec = new Ammo.btVector3(x, y, z);
231                 this.rigidBody.translate(vec);
232                 Ammo.destroy(vec);
233             },
234             /**
235              * Set posture expressed in quaternion in Rigid.
236              * @param {enchant.gl.Quat} quat
237              */
238             rotationSet: function(quat) {
239                 var qq = quat._quat;
240                 var q = new Ammo.btQuaternion(qq[0], qq[1], qq[2], -qq[3]);
241                 var t = this._getTransform();
242                 t.setRotation(q);
243                 this.rigidBody.setWorldTransform(t);
244                 Ammo.destroy(q);
245                 Ammo.destroy(t);
246             },
247             /**
248              * Applies rotation expressed in quaternion to Rigid.
249              * @param {enchant.gl.Quat} quat
250              */
251             rotationApply: function(quat) {
252                 var quat1 = quat._quat;
253                 var t = this._getTransform();
254                 var qq = t.getRotation();
255                 var quat2 = quat4.create([qq.x(), qq.y(), qq.z(), -qq.w()]);
256                 quat4.multiply(quat2, quat1, quat2);
257                 var q = new Ammo.btQuaternion(quat2[0], quat2[1], quat2[2], -quat2[3]);
258                 t.setRotation(q);
259                 this.rigidBody.setWorldTransform(t);
260                 Ammo.destroy(q);
261                 Ammo.destroy(t);
262             },
263             /**
264              */
265             clearForces: function() {
266                 var vec0 = new Ammo.btVector3(0, 0, 0);
267                 this.rigidBody.setLinearVelocity(vec0);
268                 this.rigidBody.setAngularVelocity(vec0);
269                 this.rigidBody.clearForces();
270                 Ammo.destroy(vec0);
271             },
272             /**
273              * Collision detection with another Rigid object.
274              * @param {enchant.gl.physics.Rigid} rigid Rigid to be judged.
275              * @return {Boolean} bool Presence or absence of rigid.
276              */
277             contactTest: function(rigid) {
278                 if (this.world && rigid.world
279                     && this.world == rigid.world) {
280                     return this.world.contactPairTest(this, rigid);
281                 } else {
282                     return false;
283                 }
284             },
285             /**
286              * Make Rigid valid.
287              * @param {Boolean} force Forcibly make valid.
288              */
289             activate: function(force) {
290                 this.rigidBody.activate(force);
291             },
292             /**
293              * Add power to Rigid.
294              * Power is added to the center of Rigid.
295              * @param {Number} powerX Power in x axis.
296              * @param {Number} powerY Power in y axis.
297              * @param {Number} powerZ Power in z axis.
298              */
299             applyCentralImpulse: function(powx, powy, powz) {
300                 var powv = new Ammo.btVector3(powx, powy, powz);
301                 this.activate();
302                 this.rigidBody.applyCentralImpulse(powv);
303                 Ammo.destroy(powv);
304             },
305             /**
306              * Add power to Rigid.
307              * Power is added to the specified position.
308              * @param {Number} powerX x axis power.
309              * @param {Number} powerY y axis power.
310              * @param {Number} powerZ z axis power.
311              * @param {Number} positonX Coordinate on x axis to add power to.
312              * @param {Number} positonY Coordinate on y axis to add power to.
313              * @param {Number} positonZ Coordinate on z axis to add power to.
314              */
315             applyImpulse: function(powx, powy, powz, posx, posy, posz) {
316                 var powv = new Ammo.btVector3(powx, powy, powz);
317                 var posv = new Ammo.btVector3(posx, posy, posz);
318                 this.activate();
319                 this.rigidBody.applyImpulse(powv, posv);
320                 Ammo.destroy(powv);
321                 Ammo.destroy(posv);
322             },
323             _getTransform: function() {
324                 return this.rigidBody.getWorldTransform();
325             },
326             /**
327              * Kinematize Rigid.
328              */
329             kinematize: function() {
330                 var flag = this.rigidBody.getCollisionFlags();
331                 this.rigidBody.setCollisionFlags(flag | 2);
332                 this.rigidBody.setActivationState(4);
333             },
334            /**
335             * Rigid restitutions.
336             * @type Number
337             */
338             restitution: {
339                 get: function() {
340                     return this._restitution;
341                 },
342                 set: function(n) {
343                     this._restitution = n;
344                     this.rigidBody.setRestitution(n);
345                 }
346             },
347            /**
348             * Rigid frictions.
349             * @type Number
350             */
351             friction: {
352                 get: function() {
353                     return this._friction;
354                 },
355                 set: function(n) {
356                     this._friction = n;
357                     this.rigidBody.setFriction(n);
358                 }
359             }
360         });

492              * Box shape solid object.
493              * @param {Number} scale Distance between center of box to border.

515                 addWall(s - s/8, s - s/8 - s/8, s/8, s/8, 0, s/8 - s);
516                 addWall(s - s/8, s - s/8 - s/8, s/8, -s/8, 0, s - s/8);

526         enchant.gl.physics.PhyScene3D = enchant.Class.create(enchant.gl.Scene3D, {
527             /**
528              * Scene3D containing World.
529              * As time passes, physical calculation will be applied to the Sprite3D effected by addChild.
530              * @see enchant.gl.physics.World
531              * @constructs
532              * @extends enchant.gl.Scene3D
533              */
534             initialize: function() {
535                 enchant.gl.Scene3D.call(this);
536                 var game = enchant.Game.instance;
537                 this.world = new enchant.gl.physics.World();
538                 this.isPlaying = false;
539                 this.timeStep = 1 / game.fps;
540                 this.maxSubSteps = 1;
541                 this.fixedTimeStep = 1 / 60;
542                 var that = this;
543                 this._stepping = function() {
544                     that.stepSimulation(that.timeStep, that.maxSubSteps, that.fixedTimeStep);
545                 };
546             },
547             /**
548              * Add child Sprite3D.
549              * When adding PhySprite3D, the rigid body object containing PhySprite3D will be added to the World containing PhyScene3D.
550              * @param {enchant.gl.Sprite3D|enchant.gl.physics.PhySprite3D} Sprite3D Child Sprite3D to be added.
551              */
552             addChild: function(sprite) {
553                 enchant.gl.Scene3D.prototype.addChild.call(this, sprite);
554                 if (sprite instanceof enchant.gl.physics.PhySprite3D) {
555                     this.world.addRigid(sprite.rigid);
556                 }
557             },
558             /**
559              * Delete designated child Sprite3D.
560              * When PhySprite3D is set, the rigid body object containing PhySprite3D will be deleted from the World containing PhyScene3D.
561              * @param {enchant.gl.Sprite3D|enchant.gl.physics.PhySprite3D} Sprite3D 追加する子Sprite3D.
562              */
563             removeChild: function(sprite) {
564                 enchant.gl.Scene3D.prototype.removeChild.call(this, sprite);
565                 if (sprite instanceof enchant.gl.physics.PhySprite3D) {
566                     this.world.removeRigid(sprite.rigid);
567                 }
568             },
569             /**
570              * Set the gravity of World containing PhyScene3D.
571              * @param {Number} gx x axis gravity
572              * @param {Number} gy y axis gravity
573              * @param {Number} gz z axis gravity
574              */
575             setGravity: function(x, y, z) {
576                 this.world.setGravity(x, y, z);
577             },
578             /**
579              * Proceed forward in time for World containing PhySprite3D.
580              * @param {Number} timeStep Amount of time you wish to proceed. Units are seconds.
581              * @param {Number} maxSubSteps
582              * @param {Number} fixedTimeStep
583              */
584             stepSimulation: function(timeStep, maxSubSteps, fixedTimeStep) {
585                 subStep = this.world.stepSimulation(timeStep, maxSubSteps, fixedTimeStep);
586                 var e = new enchant.Event('timestep');
587                 e.timeStep = timeStep;

926             initialize: function(nx, ny, nz, dist) {
927                 var rigid = new RigidPlane(nx, ny, nz, dist);
928                 enchant.gl.physics.PhySprite3D.call(this, rigid);
929                 this.mesh = Mesh.createPlaneXZ(50);
930                 var up = vec3.create([0, 1, 0]);
931                 var norm = vec3.create([nx, ny, nz]);
932                 var axis = vec3.create();
933                 vec3.cross(norm, up, axis);
934                 var rad = Math.acos(vec3.dot(up, norm) / (vec3.length(up) * vec3.length(norm)));
935                 var q = new Quat(axis[0], axis[1], axis[2], -rad);
936                 var vertices = new Array();
937                 for (i = 0, l = this.mesh.vertices.length; i < l; i+=3) {
938                     var x = this.mesh.vertices[i];
939                     var y = this.mesh.vertices[i+1];
940                     var z = this.mesh.vertices[i+2];
941                     var arr = q.multiplyVec3([x, y, z]);
942                     vertices.push(arr[0] + nx * dist);
943                     vertices.push(arr[1] + ny * dist);
944                     vertices.push(arr[2] + nz * dist);
945                 }
946                 this.mesh.vertices = vertices;
