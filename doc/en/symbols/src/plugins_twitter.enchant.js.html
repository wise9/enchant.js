1 /**
  2  * twitter.enchant.js
  3  * @version 0.2 (2011/07/20)
  4  * @requires enchant.js v0.4.0 or later
  5  * @requires nineleap.enchant.js v0.3 or later
  6  *
  7  * @description
  8  * enchant.js extention for 9leap.net
  9  *
 10  * @usage
 11  * EXAMPLE: use twitter icon as a sprite
 12  *
 13  * enchant();
 14  * window.onload = function() {
 15  *    var game = new Game(320, 320);
 16  *    game.twitterRequest('account/verify_credentials');
 17  *    game.onload = function() {
 18  *         var player = game.twitterAssets['account/verify_credentials'][0].toSprite(48, 48);
 19  *         player.x = 0;
 20  *         player.y = 0;
 21  *         game.rootScene.addChild(player);
 22  *     };
 23  *     game.start();
 24  * }
 25  */
 26 (function() {
 27 
 28     enchant();
 29 
 30     /**
 31      * 依存ライブラリcheck
 32      */
 33     if (enchant.nineleap.memory != undefined &&
 34         Object.getPrototypeOf(enchant.nineleap.memory) == Object.prototype) {
 35         var parentModule = enchant.nineleap.memory;
 36     } else if (enchant.nineleap != undefined &&
 37         Object.getPrototypeOf(enchant.nineleap) == Object.prototype) {
 38         var parentModule = enchant.nineleap;
 39     } else {
 40         throw new Error('Cannot load nineleap.enchant.js.');
 41     }
 42 
 43     enchant.nineleap.twitter = {};
 44 
 45     /**
 46      * @scope enchant.nineleap.twitter.Game.prototype
 47      * enchant.Game or enchant.nineleap.Game クラスを拡張したクラス
 48      */
 49     enchant.nineleap.twitter.Game = enchant.Class.create(parentModule.Game, {
 50         initialize: function(width, height) {
 51             parentModule.Game.call(this, width, height);
 52             this._twitterRequests = [];
 53             this._twitterAssets = [];
 54             this.requireAuth = true;
 55             this.authorized = true;
 56             this.twitterQueue = 0;
 57         },
 58         start: function() {
 59             if (this.twitterQueue != 0) {
 60                 if (this._twitterRequests.length) {
 61                     this.twitterAssets = {};
 62                     for (var i in this._twitterRequests) {
 63                         this.twitterAssets[this._twitterRequests[i].path] = new Array();
 64                     }
 65                     for (var i = 0, l = this._twitterRequests.length; i < l; i++) {
 66                         this._twitterRequests[i]._sendRequest();
 67                     }
 68                 }
 69                 return;
 70             }
 71             parentModule.Game.prototype.start.call(this);
 72         },
 73 
 74         /**
 75          * Twitterへのリクエストを送る。
 76          *
 77          * @param path {String} Twitter APIのパス
 78          * @param option {Object} Twitter APIへのリクエスト時に送信するオプション
 79          * @param checkError {Boolean} エラーで応答がない場合 (optional)
 80          */
 81         twitterRequest: function(path, option, checkError) {
 82             if (checkError == undefined) {
 83                 checkError = true;
 84             }
 85             if (arguments.length == 2) {
 86                 if (typeof arguments[1] == 'boolean') {
 87                     checkError = arguments[1];
 88                     option = {};
 89                 } else {
 90                     checkError = true;
 91                 }
 92             }
 93             this.twitterQueue++;
 94             var id = this._twitterRequests.length;
 95             var request = new enchant.nineleap.twitter.TwitterRequest(id, path, option, checkError);
 96             this._twitterRequests.push(request);
 97         },
 98 
 99         _setTwitterAssets: function(resBody, path, checkError) {
100             var game = enchant.Game.instance;
101             if (!(resBody instanceof Array)) {
102                 resBody = Array.prototype.slice.call(arguments);
103                 resBody = resBody.filter(function(arg) {
104                     return (arg instanceof Object);
105                 });
106             }
107             this.twitterQueue--;
108             if (resBody[0] == undefined) {
109                 this.twitterAssets[path] = [];
110             } else if ('code' in resBody[0]) {
111                 if (resBody[0].code == 401 && this.requireAuth) {
112                     window.location.replace('http://9leap.net/api/login?after_login=' + window.location.href);
113                     return;
114                 } else if (resBody[0].code == 401 && !this.requireAuth) {
115                     this.authorized = false;
116                 } else if (checkError) {
117                     alert(resBody[0].code + ' Error' + '\n Please try reload');
118                     throw new Error(resBody[0].code + ': ' + resBody[0].error);
119                 } else {
120                     this.twitterAssets[path] = [];
121                 }
122             } else {
123                 for (var i = 0, l = resBody.length; i < l; i++) {
124                     if ('name' in resBody[i]) {
125                         this.twitterAssets[path][i] = new enchant.nineleap.twitter.TwitterUserData(resBody[i]);
126                         this._twitterAssets.push(resBody[i]['profile_image_url']);
127                         if ('status' in resBody[i]) {
128                             this.twitterAssets[path][i].status = new enchant.nineleap.twitter.TwitterStatusData(resBody[i].status);
129                         }
130                     } else {
131                         this.twitterAssets[path][i] = new enchant.nineleap.twitter.TwitterStatusData(resBody[i]);
132                         this.twitterAssets[path][i].user = new enchant.nineleap.twitter.TwitterUserData(resBody[i].user);
133                         this._twitterAssets.push(resBody[i].user['profile_image_url']);
134                     }
135                 }
136             }
137             if (this.twitterQueue == 0) {
138                 this.start();
139             }
140         }
141     });
142 
143     enchant.nineleap.twitter.TwitterRequest = enchant.Class.create({
144         initialize: function(id, path, option, checkError) {
145             this.path = path;
146             this.checkError = checkError;
147             var callback = '?callback=enchant.Game.instance._twitterRequests[' + id + ']._callback';
148             var src = 'http://9leap.net/api/twitter/' + path + '.json' + callback;
149             if (option) {
150                 for (var key in option) {
151                     src += '&' + key + '=' + option[key];
152                 }
153             }
154             this.script = document.createElement('script');
155             this.script.type = 'text/javascript';
156             this.script.src = src;
157         },
158         _callback: function(resBody) {
159             enchant.Game.instance._setTwitterAssets(resBody, this.path, this.checkError);
160         },
161         _sendRequest: function() {
162             document.head.appendChild(this.script);
163         }
164     });
165 
166     enchant.nineleap.twitter.TwitterUserData = enchant.Class.create({
167         initialize: function(obj) {
168             for (var prop in obj) {
169                 this[prop] = obj[prop];
170             }
171         },
172         toSprite: function(width, height) {
173             if (arguments.length < 2) {
174                 var width = 48;
175                 var height = 48;
176             }
177             var g = enchant.Game.instance;
178             var sp = new Sprite(width, height);
179             sp.image = g.assets[this['profile_image_url']];
180             return sp;
181         }
182     });
183 
184     enchant.nineleap.twitter.TwitterStatusData = enchant.Class.create({
185         initialize: function(obj) {
186             for (var prop in obj) {
187                 this[prop] = obj[prop];
188             }
189         }
190     });
191 
192 })();
