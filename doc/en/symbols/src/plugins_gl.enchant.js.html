 66 function parseColor(arg) {
 67     var color = [];
 68     if (typeof arg == 'string') {
 69         if (arg.match(/#/)) {
 70             arg.match(/[0-9a-fA-F]{2}/g).forEach(function(n) {
 71                     color[color.length] = ('0x' + n - 0) / 255;
 72                     });
 73             color[color.length] = 1.0;
 74         } else if (arg.match(/rgba/)) {
 75             arg.match(/[0-9]{1,3},/g).forEach(function(n) {
 76                     color[color.length] = parseInt(n, 10) / 255;
 77                     });
 78             color[color.length] = parseFloat(arg.match(/[0-9]\.[0-9]{1,}/)[0]);
 79         } else if (arg.match(/rgb/)) {
 80             arg.match(/[0-9]{1,3},/g).forEach(function(n) {
 81                     color[color.length] = parseInt(n, 10) / 255;
 82                     });
 83             color[color.length] = 1.0;
 84         }
 85     } else if (arg instanceof Array) {
 86         color = arg;
 87     }
 88     return color;
 89 };

 92 function createDebugContext(context) {
 93     var ctx = {};
 94     var type = '';
 95     context.names = {};
 96     for (prop in context) {
 97         type = typeof context[prop];
 98         if (type == 'function') {
 99             ctx[prop] = (function(context, prop) {
100                 return function() {
101                     value = undefined;
102                     value = context[prop].apply(context, arguments);
103                     error = context.getError();
104                     if (error) {
105                         if (prop != 'viewport') {
106                             console.log(context.names[error] + '[' + error + ']: ' + prop);
107                             console.log(arguments);
108                         }
109                     }
110                     return value;
111                 }
112             })(context, prop);
113         } else if (type == 'number') {
114             context.names[context[prop]] = prop;
115             ctx[prop] = context[prop];
116         } else {
117             ctx[prop] = context[prop];
118         }
119     }
120     ctx.getPropName = function(index) {
121         return context.names[index];
122     };
123     return ctx;

164         var vx = p1.x + p1.parent.x - p2.x - p2.parent.x;
165         var vy = p1.y + p1.parent.y - p2.y - p2.parent.y;
166         var vz = p1.z + p1.parent.z - p2.z - p2.parent.z;
167         return (vx * vx + vy * vy + vz * vz);

169     var point2BS = function(p, bs) {
170         return (point2point(p, bs) - bs.radius * bs.radius);

172     var point2AABB = function(p, aabb) {
173         var ppx = p.x + p.parent.x;
174         var ppy = p.y + p.parent.y;
175         var ppz = p.z + p.parent.z;
176         var px = aabb.parent.x + aabb.x + aabb.scale;
177         var py = aabb.parent.y + aabb.y + aabb.scale;
178         var pz = aabb.parent.z + aabb.z + aabb.scale;
179         var nx = aabb.parent.x + (aabb.x - aabb.scale);
180         var ny = aabb.parent.y + (aabb.y - aabb.scale);
181         var nz = aabb.parent.z + (aabb.z - aabb.scale);
182         var dist = 0;
183         if (ppx < nx) {
184             dist += (ppx - nx) * (ppx - nx);
185         } else if (px < ppx) {
186             dist += (ppx - px) * (ppx - px);
187         }
188         if (ppy < ny) {
189             dist += (ppy - ny) * (ppy - ny);
190         } else if (py < ppy) {
191             dist += (ppy - py) * (ppy - py);
192         }
193         if (ppz < nz) {
194             dist += (ppz - nz) * (ppz - nz);
195         } else if (pz < ppz) {
196             dist += (ppz - pz) * (ppz - pz);
197         }
198         return dist;
