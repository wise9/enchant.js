 99 /**
100  * Globaler Export der Programmbibliotheken.
101  *
102  * Wenn keine Argument übergeben werden, werden alle Klassen die in enchant.js und in den Plugins
103  * definiert sind exportiert. Falls mehr als ein Argument übergeben wurde, werden standardmäßig nur Klassen
104  * die in enchant.js selbst definiert sind exporitert. Wenn auch Plugin Klassen exportiert werden sollen,
105  * müssen die Plugin Bezeichner explizit als Argumente übergeben werden.
106  *
107  * @example
108  *   enchant();     // alle Klassen werden exportiert.
109  *   enchant('');   // nur Klassen die in enchant.js definiert sind werden exportiert.
110  *   enchant('ui'); // enchant.js Klassen und ui.enchant.js Klassen werden exportiert.
111  *
112  * @param {...String} [modules] Module die exportiert werden sollen.
113  */
114 var enchant = function(modules) {
115     if (modules != null) {
116         if (!(modules instanceof Array)) {
117             modules = Array.prototype.slice.call(arguments);
118         }
119         modules = modules.filter(function(module) {
120             return [module].join();
121         });
122     }
123     (function include(module, prefix) {
124         var submodules = [],
125             i, len;
126         for (var prop in module){
127             if (module.hasOwnProperty(prop)) {
128                 if (typeof module[prop] === 'function') {
129                     window[prop] = module[prop];
130                 } else if (typeof module[prop] === 'object' && Object.getPrototypeOf(module[prop]) === Object.prototype) {
131                     if (modules == null) {
132                         submodules.push(prop);
133                     } else {
134                         i = modules.indexOf(prefix + prop);
135                         if (i !== -1) {
136                             submodules.push(prop);
137                             modules.splice(i, 1);
138                         }
139                     }
140                 }
141             }
142         }
143 
144         for (i = 0, len = submodules.length; i < len; i++) {
145             include(module[submodules[i]], prefix + submodules[i] + '.');
146         }
147     }(enchant, ''));
148 
149     if (modules != null && modules.length) {
150         throw new Error('Cannot load module: ' + modules.join(', '));
151     }
152 };

195 /**
196  * Erstellt eine neue Klasse
197  *
198  * Wenn eine Klasse definiert wird, die von einer anderen Klasse erbt, wird der Konstruktor der
199  * Basisklasse als Standard definiert. Sollte dieser Konstruktor in der neuen Klasse überschrieben
200  * werden, sollte der vorherige Konstruktor explizit aufgerufen werden, um eine korrekte
201  * Klasseninitialisierung sicherzustellen.
202  * 
203  * @example
204  *   var Ball = Class.create({ // definiert eine unabhängige Klasse.
205  *       initialize: function(radius) { ... }, // Methodendefinitionen
206  *       fall: function() { ... }
207  *   });
208  *
209  *   var Ball = Class.create(Sprite);  // definiert eine Klasse die von "Sprite" erbt.
210  *   var Ball = Class.create(Sprite, { // definiert eine Klasse die von "Sprite" erbt.
211  *       initialize: function(radius) { // überschreibt den Standardkonstruktor.
212  *          Sprite.call(this, radius*2, radius*2); // Aufruf des Konstruktors der Basisklasse.
213  *          this.image = game.assets['ball.gif'];
214  *       }
215  *   });
216  *
217  * @param {Function} [superclass] Die Klasse, deren Klassendefinition
218  * die neue Klasse erben wird.
219  * @param {*} definition Klassendefinition.
220  * @static
221  */
222 enchant.Class.create = function(superclass, definition) {
223     if (arguments.length === 0) {
224         return enchant.Class.create(Object, definition);
225     } else if (arguments.length === 1 && typeof arguments[0] !== 'function') {
226         return enchant.Class.create(Object, arguments[0]);
227     }
228 
229     for (var prop in definition) {
230         if (definition.hasOwnProperty(prop)) {
231             if (typeof definition[prop] === 'object' && Object.getPrototypeOf(definition[prop]) === Object.prototype) {
232                 if (!('enumerable' in definition[prop])) {
233                     definition[prop].enumerable = true;
234                 }
235             } else {
236                 definition[prop] = { value: definition[prop], enumerable: true, writable: true };
237             }
238         }
239     }
240     var Constructor = function() {
241         if (this instanceof Constructor) {
242             Constructor.prototype.initialize.apply(this, arguments);
243         } else {
244             return new Constructor();
245         }
246     };
247     Constructor.prototype = Object.create(superclass.prototype, definition);
248     Constructor.prototype.constructor = Constructor;
249     if (Constructor.prototype.initialize == null) {
250         Constructor.prototype.initialize = function() {
251             superclass.apply(this, arguments);
252         };
253     }
254 
255     return Constructor;
256 };
