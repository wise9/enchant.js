195 var parseColor = function(string) {
196     var color = [];
197     if (typeof string == 'string') {
198         if (string.match(/#/)) {
199             string.match(/[0-9a-fA-F]{2}/g).forEach(function(n) {
200                 color[color.length] = ('0x' + n - 0) / 255;
201             });
202             color[color.length] = 1.0;
203         } else if (string.match(/rgba/)) {
204             string.match(/[0-9]{1,3},/g).forEach(function(n) {
205                 color[color.length] = parseInt(n, 10) / 255;
206             });
207             color[color.length] = parseFloat(string.match(/[0-9]\.[0-9]{1,}/)[0]);
208         } else if (string.match(/rgb/)) {
209             string.match(/[0-9]{1,3},/g).forEach(function(n) {
210                 color[color.length] = parseInt(n, 10) / 255;
211             });
212             color[color.length] = 1.0;
213         }
214     } else if (string instanceof Array) {
215         color = string;
216     }
217     return color;
218 };

220 var createDebugContext = function(context) {
221     var ctx = {};
222     var names = {};
223     var type = '';
224     var val;
225     for (prop in context) {
226         type = typeof context[prop];
227         val = context[prop];
228         if (type == 'function') {
229             ctx[prop] = (function(context, prop) {
230                 return function() {
231                     var value, error;
232                     value = context[prop].apply(context, arguments);
233                     error = context.getError();
234                     if (error) {
235                         console.log(names[error] + '(' + error + ')' + ': ' + prop);
236                         console.log(arguments);
237                     }
238                     return value;
239                 }
240             })(context, prop);
241         } else if (type == 'number') {
242             names[val] = prop;
243             ctx[prop] = val;
244         } else {
245             ctx[prop] = val;
246         }
247     }
248     ctx.getNameById = function(i) {
249         return names[i];
250     };
251     return ctx;

273 var TextureManager = enchant.Class.create({
274     initialize: function() {
275         this.storage = {};
276     },
277     hasTexture: function(src) {
278         return src in this.storage;
279     },
280     getWebGLTexture: function(image, flip, wrap, mipmap) {
281         var ret;
282         if (this.hasTexture(image.src)) {
283             ret = this.storage[image.src];
284         } else {
285             ret = this.createWebGLTexture(image, flip, wrap, mipmap);
286         }
287         return ret;
288     },
289     isPowerOfTwo: function(n) {
290         return (n > 0) && ((n & (n - 1)) == 0);
291     },
292     setTextureParamater: function(power, target, wrap, mipmap) {
293         var filter;
294         if (mipmap) {
295             filter = gl.LINEAR_MIPMAP_LINEAR;
296         } else {
297             filter = gl.NEAREST;
298         }
299         if (!power) {
300             wrap = gl.CLAMP_TO_EDGE;
301         }
302         gl.texParameteri(target, gl.TEXTURE_WRAP_S, wrap);
303         gl.texParameteri(target, gl.TEXTURE_WRAP_T, wrap);
304         gl.texParameteri(target, gl.TEXTURE_MAG_FILTER, filter);
305         gl.texParameteri(target, gl.TEXTURE_MIN_FILTER, filter);
306     },
307     _texImage: function(image, target) {
308         gl.texImage2D(target, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, image);
309     },
310     _writeWebGLTexture: function(image, target, wrap, mipmap) {
311         var power = this.isPowerOfTwo(image.width) && this.isPowerOfTwo(image.height);
312         if (typeof target == 'undefined') {
313             target = gl.TEXTURE_2D;
314         }
315         if (typeof wrap == 'undefined') {
316             wrap = gl.REPEAT;
317         }
318         this.setTextureParamater(power, target, wrap, mipmap);
319 
320         this._texImage(image, target);
321         if (mipmap) {
322             gl.generateMipmap(target);
323         }
324     },
325     createWebGLTexture: function(image, flip, wrap, mipmap) {
326         var tex = gl.createTexture();
327         var target = gl.TEXTURE_2D;
328         gl.bindTexture(target, tex);
329         gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, flip);
330         this._writeWebGLTexture(image, target, wrap, mipmap);
331         gl.bindTexture(target, null);
332         this.storage[image.src] = tex;
333         return tex;
334     },
335     createWebGLCubeMapTexture: function(images, wrap, mipmap) {
336         var faceTargets = [
337             gl.TEXTURE_CUBE_MAP_POSITIVE_X,
338             gl.TEXTURE_CUBE_MAP_NEGATIVE_X,
339             gl.TEXTURE_CUBE_MAP_POSITIVE_Y,
340             gl.TEXTURE_CUBE_MAP_POSITIVE_Y,
341             gl.TEXTURE_CUBE_MAP_NEGATIVE_Z,
342             gl.TEXTURE_CUBE_MAP_NEGATIVE_Z
343         ];
344 
345         var tex = gl.createTexture();
346         var target, image;
347         gl.bindTexture(gl.TEXTURE_CUBE_MAP, tex);
348         gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, true);
349         for (var i = 0, l = images.length; i < l; i++) {
350             target = faceTargets[i];
351             image = images[i];
352             this._writeWebGLTexture(image, target, wrap, mipmap);
353         }
354         gl.bindTexture(gl.TEXTURE_CUBE_MAP, null);
355         return tex;
356     }

449      * WebGLのシェーダプログラムを管理するクラス.

501         }

504      * シェーダプログラムをコンパイルする.
505      * コンストラクタからシェーダソースを渡した場合は自動的にコンパイルされる.
506      * @example
507      * var shader = new Shader();
508      * // シェーダプログラムのソースを渡す.
509      * shader.vShaderSource = vert;
510      * shader.fShaderSource = frag;
511      * // コンパイル.
512      * shader.compile();
513      */
514     compile: function() {
515         if (this._updatedVShaderSource) {
516             this._prepareVShader();
517         }
518         if (this._updatedFShaderSource) {
519             this._prepareFShader();
520         }
521         if (this._program == null) {
522             this._program = gl.createProgram();
523         } else {
524             gl.detachShader(this._program, this._vShaderProgram);
525             gl.detachShader(this._program, this._fShaderProgram);
526         }
527         gl.attachShader(this._program, this._vShaderProgram);
528         gl.attachShader(this._program, this._fShaderProgram);
529         gl.linkProgram(this._program);
530         if (!gl.getProgramParameter(this._program, gl.LINK_STATUS)) {
531             this._logShadersInfo();
532             throw 'could not compile shader';
533         }
534         this._getAttributesProperties();
535         this._getUniformsProperties();
536     },
537     /**
538      * シェーダプログラムを使用するように設定する.
539      */
540     use: function() {
541         gl.useProgram(this._program);
542     },
543     /**
544      * シェーダプログラムにattribute変数をセットする.
545      * enchant.gl.Sprite3Dの内部などで使用される.

558     },

568      * });

572             this._uniforms[prop] = params[prop];
573         }
574     },

579         gl.shaderSource(this._vShaderProgram, this._vShaderSource);

581         this._updatedVShaderSource = false;

594     },
595     _getAttributesProperties: function() {
596         var n, info;
597         n = gl.getProgramParameter(this._program, gl.ACTIVE_ATTRIBUTES);
598         for (var i = 0; i < n; i++) {
599             var info = gl.getActiveAttrib(this._program, i);

608             var info = gl.getActiveUniform(this._program, i);
609             this._uniforms[info.name];
610             addUniformsProperty(this, info);
611         }
612     },
613     /**
614      * オブジェクトを破棄する.
615     */

617         gl.deleteProgram(this._vShaderProgram);

621     }

641 var addUniformsProperty = function(program, info) {
642     var name = info.name;
643     var loc = gl.getUniformLocation(program._program, info.name);
644     var suffix;
645     var sampler = false;
646     var matrix = false;
647     var desc = {
648         get: function() {
649             return 'uniform';
650         }
651     };
652     switch(info.type) {
653         case gl.FLOAT:
654             suffix = '1f';
655             break;
656 
657         case gl.FLOAT_MAT2:
658             matrix = true;
659         case gl.FLOAT_VEC2:
660             suffix = '2fv';
661             break;
662 
663         case gl.FLOAT_MAT3:
664             matrix = true;
665         case gl.FLOAT_VEC3:
666             suffix = '3fv';
667             break;
668 
669         case gl.FLOAT_MAT4:
670             matrix = true;
671         case gl.FLOAT_VEC4:
672             suffix = '4fv';
673             break;
674 
675         case gl.SAMPLER_2D:
676         case gl.SAMPLER_CUBE:
677             sampler = true;
678         case gl.INT:
679         case gl.INT:
680         case gl.BOOL:
681             suffix = '1i';
682             break;
683 
684         case gl.INT_VEC2:
685         case gl.BOOL_VEC2:
686             suffix = '2iv';
