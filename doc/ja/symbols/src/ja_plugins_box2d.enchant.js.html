 61     enchant.box2d.STATIC_SPRITE = 0;

 68     /**
 69      * @scope enchant.box2d.PhysicsWorld.prototype
 70      */
 71     enchant.box2d.PhysicsWorld = enchant.Class.create({
 72         /**
 73          * 物理シミュレーションを行う世界のクラス
 74          * @example
 75          *   //y軸方向へ重力加速度9.8m/s^2
 76          *   var physicsWorld = new PhysicsWorld(0, 9.8);
 77          *   //無重力
 78          *   var physicsWorld = new PhysicsWorld(0, 0);
 79          *
 80          * @param {Number} [gravityX] x軸方向への引力.
 81          * @param {Number} [gravityY] y軸方向への引力.
 82          * @constructs
 83          */
 84         initialize: function (gravityX, gravityY) {
 85             /**
 86              * 物理シミュレーションの精度
 87              * @type {Nunber}
 88              */
 89             this.iterations = 10;
 90             world = new b2World(
 91                 new b2Vec2(gravityX, gravityY)  //gravity
 92                 , true                          //allow sleep
 93             );
 94         },
 95         /**
 96          * 物理シミュレーション内の時間を進める
 97          * @param {b2Vec2} [pos] Spriteの座標.
 98          */
 99         step: function (fps) {
100             world.Step(1 / fps, this.iterations, this.iterations);
101         },
102         /**
103          * 物体の当たり判定
104          * @example
105          *   //ぶつかった2つのSpriteを消す
106          *   physicsWorld.contact(function (sprite1, sprite2) {
107          *       sprite1.destroy();
108          *       sprite2.destroy();
109          *   });
110          *
111          * @param {function(sprite1:enchant.box2d.PhySprite,sprite2:enchant.box2d.PhySprite)} [func] 当たり判定時の処理
112          */
113         contact: function (func) {
114             var c = world.m_contactList;
115             if(c) {
116                 for (var contact = c; contact; contact = contact.m_next) {
117                     var pos1 = contact.m_fixtureA.m_body.GetPosition().Copy();
118                     pos1.Subtract(contact.m_fixtureB.m_body.GetPosition());
119                     pos1.Multiply(WORLD_SCALE);
120                     var r1 = (contact.m_fixtureA.m_body.m_userData.width + contact.m_fixtureB.m_body.m_userData.width) / 2;
121                     var r2 = (contact.m_fixtureA.m_body.m_userData.height + contact.m_fixtureB.m_body.m_userData.height) / 2;
122                     if(Math.abs(pos1.x) <= r1 && Math.abs(pos1.y) <= r2) {
123                         func(contact.m_fixtureA.m_body.m_userData, contact.m_fixtureB.m_body.m_userData);
124                     }
125                 }
126             }
127         }
128     });

131     /**
132      * @scope enchant.box2d.PhySprite.prototype
133      */
134     enchant.box2d.PhySprite = enchant.Class.create(enchant.Sprite, {
135         /**
136          * 画像表示機能を持った物理シミュレーションクラス.
137          * @param {Number} [width] Spriteの横幅.
138          * @param {Number} [height] Spriteの高さ.
139          * @constructs
140          * @extends enchant.Sprite
141          */
142         initialize: function (width, height) {
143             this.body;
144             /**
145              * 静的オブジェクトか動的オブジェクトか
146              */
147             this.staticOrDynamic;
148             enchant.Sprite.call(this, width, height);
149 
150             var time = 0;
151             this.addEventListener(enchant.Event.ENTER_FRAME, function (e) {
152                 this.x = this.x;
153                 this.y = this.y;
154                 if(time % 2) {   //なぜか移動と回転を一緒にできない。謎。
155                     this.rotation = this.angle;
156                 }
157                 time++;
158                 time = time % 2;
159             });
160         },
161         /**
162          * 四角形の物理シミュレーション用Sprite生成.
163          * @param {Boolean} staticOrDynamic 静止するか動くか.
164          * @param {Number} density Spriteの密度.
165          * @param {Number} friction Spriteの摩擦.
166          * @param {Number} restitution Spriteの反発.
167          * @param {Boolean} isSleeping Spriteが初めから物理演算を行うか.
168          */
169         createPhyBox: function (staticOrDynamic, density, friction, restitution, awake) {
170             this.staticOrDynamic = staticOrDynamic;
171             var fixDef = new b2FixtureDef;
172             fixDef.density = (density != null ? density : 1.0);             // 密度
173             fixDef.friction = (friction != null ? friction : 0.5);          // 摩擦
174             fixDef.restitution = (restitution != null ? restitution : 0.3); // 反発
175             fixDef.shape = new b2PolygonShape;
176             fixDef.shape.SetAsBox(this.width / 2 / WORLD_SCALE, this.height / 2 / WORLD_SCALE);
177             var bodyDef = new b2BodyDef;
178             bodyDef.type = staticOrDynamic;
179             bodyDef.position.x = 0;
180             bodyDef.position.y = 0;
181             bodyDef.awake = (awake != null ? awake : true);
182             bodyDef.userData = this;
183             return world.CreateBody(bodyDef).CreateFixture(fixDef);
184         },
185         /**
186          * 円形の物理シミュレーション用Sprite生成.
187          * @param {Boolean} staticOrDynamic 静止するか動くか.
188          * @param {Number} density Spriteの密度.
189          * @param {Number} friction Spriteの摩擦.
190          * @param {Number} restitution Spriteの反発.
191          * @param {Boolean} isSleeping Spriteが初めから物理演算を行うか.
192          */
193         createPhyCircle: function (staticOrDynamic, density, friction, restitution, awake) {
194             this.staticOrDynamic = staticOrDynamic;
195             var fixDef = new b2FixtureDef;
196             fixDef.density = (density != null ? density : 1.0);             // 密度
197             fixDef.friction = (friction != null ? friction : 0.5);          // 摩擦
198             fixDef.restitution = (restitution != null ? restitution : 0.3); // 反発
199             fixDef.shape = new b2CircleShape(this.width / 2 / WORLD_SCALE);
200             var bodyDef = new b2BodyDef;
201             bodyDef.type = staticOrDynamic;
202             bodyDef.position.x = 0;
203             bodyDef.position.y = 0;
204             bodyDef.awake = (awake != null ? awake : true);
205             bodyDef.userData = this;
206             return world.CreateBody(bodyDef).CreateFixture(fixDef);
207         },
208         /**
209          * Spriteのタイプ 静的（STATIC_SPRITE）か動的（DYNAMIC_SPRITE)か
210          * @type {bool}
211          */
212         type: {
213             get: function () {
214                 if(this.body.m_body.GetType() == b2Body.b2_staticBody)
215                     return enchant.box2d.STATIC_SPRITE;
216                 return enchant.box2d.DYNAMIC_SPRITE;
217             },
218             set: function (staticOrDynamic) {
219                 this.staticOrDynamic = staticOrDynamic;
220                 this.body.m_body.SetType(staticOrDynamic);
221             }
222         },
223         /**
224          * Spriteのx座標.
225          * @type {Number}
226          */
227         x: {
228             get: function () {
229                 return this.body.m_body.GetPosition().x * WORLD_SCALE - this.width / 2;
230             },
231             set: function (x) {
232                 this._x = x;
233                 x += this.width / 2;
234                 this.body.m_body.SetPosition(new b2Vec2(x / WORLD_SCALE, this.body.m_body.GetPosition().y));
235                 this._updateCoordinate();
236             }
237         },
238         /**
239          * Spriteのy座標.
240          * @type {Number}
241          */
242         y: {
243             get: function () {
244                 return this.body.m_body.GetPosition().y * WORLD_SCALE - this.height / 2;
245             },
246             set: function (y) {
247                 this._y = y;
248                 y += this.height / 2;
249                 this.body.m_body.SetPosition(new b2Vec2(this.body.m_body.GetPosition().x, y / WORLD_SCALE));
250                 this._updateCoordinate();
251             }
252         },
253         /**
254          * Spriteの中心のx座標.
255          * @type {Number}
256          */
257         centerX: {
258             get: function () {
259                 return this.x + this.width / 2;
260             },
261             set: function (x) {
262                 this.x = x - this.width / 2;
263             }
264         },
265         /**
266          * Spriteの中心のy座標.
267          * @type {Number}
268          */
269         centerY: {
270             get: function () {
271                 return this.y + this.height / 2;
272             },
273             set: function (y) {
274                 this.y = y - this.height / 2;
275             }
276         },
277         /**
278          * Spriteの中心座標ベクトル.
279          * @type {b2Vec2}
280          */
281         position: {
282             get: function () {
283                 var pos = this.body.m_body.GetPosition().Copy();
284                 pos.Multiply(WORLD_SCALE);
285                 return pos;
286             },
287             set: function (pos) {
288                 this.centerX = pos.x;
289                 this.centerY = pos.y;
290                 this.body.m_body.SetPosition(new b2Vec2(pos.x / WORLD_SCALE, pos.y / WORLD_SCALE));
291             }
292         },
293         /**
294          * Spriteのx座標の速度（単位はpx/s）.
295          * @type {Number}
296          */
297         vx: {
298             get: function () {
299                 return this.body.m_body.GetLinearVelocity().x * WORLD_SCALE;
300             },
301             set: function (x) {
302                 this.body.m_body.SetLinearVelocity(new b2Vec2(x / WORLD_SCALE, this.body.m_body.GetLinearVelocity().y));
303             }
304         },
305         /**
306          * Spriteのy座標の速度（単位はpx/s）.
307          * @type {Number}
308          */
309         vy: {
310             get: function () {
311                 return this.body.m_body.GetLinearVelocity().y * WORLD_SCALE;
312             },
313             set: function (y) {
314                 this.body.m_body.SetLinearVelocity(new b2Vec2(this.body.m_body.GetLinearVelocity().x, y / WORLD_SCALE));
315             }
316         },
317         /**
318          * Spriteの速度（単位はpx/s）.
319          * @type {b2Vec2}
320          */
321         velocity: {
322             get: function () {
323                 var v = this.body.m_body.GetLinearVelocity().Copy();
324                 v.Multiply(WORLD_SCALE);
325                 return v;
326             },
327             set: function (v) {
328                 this.body.m_body.SetLinearVelocity(new b2Vec2(v.x / WORLD_SCALE, v.y / WORLD_SCALE));
329             }
330         },
331         /**
332          * Spriteの角度 (度数法)..
333          * @type {Number}
334          */
335         angle: {
336             get: function () {
337                 return this.body.m_body.GetAngle() * (180 / Math.PI);
338             },
339             set: function (angle) {
340                 this.rotation = angle;
341                 this.body.m_body.SetAngle(angle * (Math.PI / 180));
342             }
343         },
344         /**
345          * Spriteの角速度（単位はdeg/s）.
346          * @type {b2Vec2}
347          */
348         angularVelocity: {
349             get: function () {
350                 return this.body.m_body.GetAngularVelocity() * (180 / Math.PI);
351             },
352             set: function (omega) {
353                 this.setAwake(true);
354                 this.body.m_body.SetAngularVelocity(omega * (Math.PI / 180));
355             }
356         },
357         /**
358          * 継続的な力を加える
359          * @param {b2Vec2} force 加える力のベクトル
360          */
361         applyForce: function (force) {
362             this.setAwake(true);
363             this.body.m_body.ApplyForce(force, this.body.m_body.GetPosition());
364         },
365         /**
366          * 瞬間的な力を加える
367          * @param {b2Vec2} impulse 加える力のベクトル
368          */
369         applyImpulse: function (impulse) {
370             this.setAwake(true);
371             this.body.m_body.ApplyImpulse(impulse, this.body.m_body.GetPosition());
372         },
373         /**
374          * 継続的な回転力を与える
375          * @param {Number} torque 加える回転力
376          */
377         applyTorque: function (torque) {
378             this.setAwake(true);
379             this.body.m_body.ApplyTorque(torque);
380         },
381         /**
382          * 物理シミュレーションされているか
383          * @type {Boolean}
384          */
385         sleep: {
386             get: function () {
387                 return this.body.m_body.IsSleepingAllowed();
388             },
389             set: function (flag) {
390                 this.setAwake(true);
391                 this.body.m_body.SetSleepingAllowed(flag);
392             }
393         },
394         /**
395          * 物理シミュレーションされていない時、物理シミュレーションを行う(sleep時は動かなくなるので)
396          * @param {Boolean} flag 物理シミュレーションを行うかどうか
397          */
398         setAwake: function (flag) {
399             this.body.m_body.SetAwake(flag);
400         },
401         /**
402          * 衝突判定
403          * @example
404          *   //bearに当たったSpriteを消す
405          *   bear.contact(function (sprite) {
406          *      sprite.destroy();
407          *   });
408          *
409          * @param {function(sprite:enchant.box2d.PhySprite)} [func] ぶつかったSpriteを引数とする関数
410          */
411         contact: function (func) {
412             var c = this.body.m_body.m_contactList;
413             if(c) {
414                 for (var contact = c.contact; contact; contact = contact.m_next) {
415                     var pos1 = contact.m_fixtureA.m_body.GetPosition().Copy();
416                     pos1.Subtract(contact.m_fixtureB.m_body.GetPosition());
417                     pos1.Multiply(WORLD_SCALE);
418                     var r1 = (contact.m_fixtureA.m_body.m_userData.width + contact.m_fixtureB.m_body.m_userData.width) / 1.5;
419                     var r2 = (contact.m_fixtureA.m_body.m_userData.height + contact.m_fixtureB.m_body.m_userData.height) / 1.5;
420                     if(Math.abs(pos1.x) <= r1 && Math.abs(pos1.y) <= r2) {
421                         //片方が自分ならもう片方をぶつかった相手として処理する
422                         if(this.body.m_body == contact.m_fixtureA.m_body)
423                             func(contact.m_fixtureB.m_body.m_userData);
424                         else if(this.body.m_body == contact.m_fixtureB.m_body)
425                             func(contact.m_fixtureA.m_body.m_userData);
426                     }
427                 }
428             }
429         },
430         /**
431          * 物体の削除
432          * removeChildではなくこちらでSpriteを取り除く
433          */
434         destroy: function () {
435             if(this.scene != null) {
436                 world.DestroyBody(this.body.m_body);
437                 this.body.Destroy();
438                 this.scene.removeChild(this);
439             }
440         }
441 
442     });

448         /**
