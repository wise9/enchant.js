 22         /**
 23          * @scope enchant.gl.physics.World.prototype
 24          */
 25         enchant.gl.physics.World = enchant.Class.create({
 26             /**
 27              * 物理演算が適用される世界.
 28              * ここに剛体オブジェクトを追加し, 時間を進めることで物理演算が実行される.
 29              * @see enchant.gl.physics.PhyScene3D
 30              * @constructs
 31              */
 32             initialize: function() {
 33                 var g = new Ammo.btVector3(0, -10, 0);
 34                 var collisionConfiguration = new Ammo.btDefaultCollisionConfiguration();
 35                 var dispatcher = new Ammo.btCollisionDispatcher(collisionConfiguration);
 36                 var pairCache = new Ammo.btDbvtBroadphase();
 37                 var constraintSolver = new Ammo.btSequentialImpulseConstraintSolver();
 38 
 39                 this._dynamicsWorld = new Ammo.btDiscreteDynamicsWorld(
 40                     dispatcher, pairCache, constraintSolver, collisionConfiguration);
 41 
 42                 this._dynamicsWorld.setGravity(g);
 43 
 44                 Ammo.destroy(g);
 45             },
 46             /**
 47              * Worldの重力を設定する.
 48              * @param {Number} gx x軸方向の重力.
 49              * @param {Number} gy y軸方向の重力.
 50              * @param {Number} gz z軸方向の重力.
 51              */
 52             setGravity: function(gx, gy, gz) {
 53                 var g = new Ammo.btVector3(gx, gy, gz);
 54                 this._dynamicsWorld.setGravity(g);
 55                 Ammo.destroy(g);
 56             },
 57             /**
 58              * Worldの時間を進める.
 59              * timeStepがfixedTimeStepより大きい場合, maxSubStepで指定した回数まで続けて時間を進める.
 60              * @param {Number} timeStep 進めたい時間.単位は秒.
 61              * @param {Number} maxSubSteps シミュレーションの最大追加回数.
 62              * @param {Number} fixedTimeStep 基本となる時間. デフォルト値は1/60.
 63              * @return {Number} subStepsNum
 64              */
 65             stepSimulation: function(timeStep, maxSubSteps, fixedTimeStep) {
 66                 return this._dynamicsWorld.stepSimulation(timeStep, maxSubSteps, fixedTimeStep);
 67             },
 68             /**
 69              * Worldに剛体を追加する.
 70              * @param {enchant.gl.physics.Rigid} Rigid 追加する剛体オブジェクト.
 71              */
 72             addRigid: function(rigid) {
 73                 this._dynamicsWorld.addRigidBody(rigid.rigidBody);
 74                 rigid.world = this;
 75             },
 76             /**
 77              * Worldから剛体を削除する.
 78              * @param {enchant.gl.physics.Rigid} Rigid 削除する剛体オブジェクト.
 79              */
 80             removeRigid: function(rigid) {
 81                 this._dynamicsWorld.removeRigidBody(rigid.rigidBody);
 82                 rigid.world = null;
 83             },
 84             /**
 85              * Rigid同士が衝突しているかを判定する.
 86              * @param {enchant.gl.physics.Rigid} rigid1 判定するRigid1.
 87              * @param {enchant.gl.physics.Rigid} rigid2 判定するRigid2.
 88              * @return {Boolean} bool 衝突の有無.
 89              */
 90             contactPairTest: function(rigid1, rigid2) {
 91                 var callback = new Ammo.ConcreteContactResultCallback();
 92                 var result = false;
 93                 Ammo.customizeVTable(callback, [{
 94                     original: Ammo.ConcreteContactResultCallback.prototype.addSingleResult,
 95                     replacement: function(tp, cp, colObj0, partid0, index0, colObj1, partid1, index1) {
 96                         result = true;
 97                     }
 98                 }]);
 99                 this._dynamicsWorld.contactPairTest(rigid1.rigidBody, rigid2.rigidBody, callback);
100                 Ammo.destroy(callback);
101                 return result;
102             }
103         });

105         /**
106          * @scope enchant.gl.physics.Rigid.prototype
107          */
108         enchant.gl.physics.Rigid = enchant.Class.create({
109             /**
110              * 剛体オブジェクト.
111              * Worldに追加して使用する.
112              * @param shape Ammo.btCollisionShapeオブジェクト.
113              * @param {Number} mass 剛体の質量.
114              * @param {Number} linearDamping 剛体の線形速度の減衰率.
115              * @param {Number} angularDamping 剛体の角速度の減衰率.
116              * @see enchant.gl.physics.RigidBox
117              * @see enchant.gl.physics.RigidCube
118              * @see enchant.gl.physics.RigidSphere
119              * @see enchant.gl.physics.RigidCylinder
120              * @see enchant.gl.physics.RigidCapsule
121              * @see enchant.gl.physics.RigidPlane
122              * @see enchant.gl.physics.RigidContainer
123              * @constructs
124              */
125             initialize: function(shape, mass, lDamp, aDamp) {
126                 if (typeof shape == 'undefined') {
127                     shape = new Ammo.btBoxShape(1);
128                 } if (typeof mass == 'undefined') {
129                     mass = 1;
130                 }
131 
132                 var localInertia = new Ammo.btVector3(0, 0, 0);
133                 shape.calculateLocalInertia(mass, localInertia);
134 
135                 var transform = new Ammo.btTransform();
136                 transform.setIdentity();
137 
138                 var motionState = new Ammo.btDefaultMotionState(transform);
139                 var rigidBodyInfo = new Ammo.btRigidBodyConstructionInfo(mass, motionState, shape, localInertia);
140                 rigidBodyInfo.set_m_restitution(0.1);
141                 rigidBodyInfo.set_m_friction(3.0);
142 
143                 if (typeof lDamp != 'undefined') {
144                     rigidBodyInfo.set_m_linearDamping(lDamp);
145                 } if (typeof aDamp != 'undefined') {
146                     rigidBodyInfo.set_m_angularDamping(aDamp);
147                 }
148 
149                 this.shape = shape;
150                 /**
151                  * Rigidが所属するWorld
152                  */
153                 this.world = null;
154 
155                 /**
156                  * Ammoの剛体オブジェクト
157                  */
158                 this.rigidBody = new Ammo.btRigidBody(rigidBodyInfo);
159                 var p = Ammo.getPointer(this.rigidBody);
160                 enchant.gl.physics.Rigid._refs[p] = this;
161 
162                 Ammo.destroy(transform);
163                 Ammo.destroy(localInertia);
164                 Ammo.destroy(rigidBodyInfo);
165 
166                 this._x = 0;
167                 this._y = 0;
168                 this._z = 0;
169                 this._scaleX = 1;
170                 this._scaleY = 1;
171                 this._scaleZ = 1;
172                 this._mass = mass;
173                 this._restitution = 0.3;
174                 this._friction = 0.3;
175             },
176             /**
177              * Rigidを拡大縮小する.
178              * Worldでの現在の拡大率から, 各軸に対して指定された倍率分だけ拡大縮小をする.
179              * @param {Number} x x軸方向の拡大率.
180              * @param {Number} y y軸方向の拡大率.
181              * @param {Number} z z軸方向の拡大率.
182              */
183             scale: function(x, y, z) {
184                 this.activate();
185                 this._scaleX *= x;
186                 this._scaleY *= y;
187                 this._scaleZ *= z;
188                 var scale = new Ammo.btVector3(this._scaleX, this._scaleY, this._scaleZ);
189                 this.shape.setLocalScaling(scale);
190                 Ammo.destroy(scale);
191             },
192             _scaleAxis: function(axis, scale) {
193                 axis.toUpperCase();
194                 this['_scale' + axis] = scale;
195                 var scale = new Ammo.btVector3(this._scaleX, this._scaleY, this._scaleZ);
196                 this.shape.setLocalScaling(scale);
197                 Ammo.destroy(scale);
198             },
199             /**
200              * Rigidを平行移動する.
201              * Worldでの現在の位置から, 各軸に対して指定された分だけ平行移動をする.
202              * @param {Number} x x軸方向の平行移動量.
203              * @param {Number} y y軸方向の平行移動量.
204              * @param {Number} z z軸方向の平行移動量.
205              */
206             translate: function(x, y, z) {
207                 this.activate();
208                 var vec = new Ammo.btVector3(x, y, z);
209                 this.rigidBody.translate(vec);
210                 Ammo.destroy(vec);
211                 this._x += x;
212                 this._y += y;
213                 this._z += z;
214             },
215             _translateAxis: function(axis, n) {
216                 this.activate();
217                 var x = 0;
218                 var y = 0;
219                 var z = 0;
220                 if (axis == 'x') {
221                     x = n - this._x;
222                     this._x = n;
223                 } else if (axis == 'y') {
224                     y = n - this._y;
225                     this._y = n;
226                 } else if (axis == 'z') {
227                     z = n - this._z;
228                     this._z = n;
229                 }
230                 var vec = new Ammo.btVector3(x, y, z);
231                 this.rigidBody.translate(vec);
232                 Ammo.destroy(vec);
233             },
234             /**
235              * クォータニオンで表した姿勢をRigidにセットする.
236              * @param {enchant.gl.Quat} quat
237              */
238             rotationSet: function(quat) {
239                 var qq = quat._quat;
240                 var q = new Ammo.btQuaternion(qq[0], qq[1], qq[2], -qq[3]);
241                 var t = this._getTransform();
242                 t.setRotation(q);
243                 this.rigidBody.setWorldTransform(t);
244                 Ammo.destroy(q);
245                 Ammo.destroy(t);
246             },
247             /**
248              * クォータニオンで表した回転をRigidに適用する.
249              * @param {enchant.gl.Quat} quat
250              */
251             rotationApply: function(quat) {
252                 var quat1 = quat._quat;
253                 var t = this._getTransform();
254                 var qq = t.getRotation();
255                 var quat2 = quat4.create([qq.x(), qq.y(), qq.z(), -qq.w()]);
256                 quat4.multiply(quat2, quat1, quat2);
257                 var q = new Ammo.btQuaternion(quat2[0], quat2[1], quat2[2], -quat2[3]);
258                 t.setRotation(q);
259                 this.rigidBody.setWorldTransform(t);
260                 Ammo.destroy(q);
261                 Ammo.destroy(t);
262             },
263             /**
264              * Rigidを止める.
265              * Stops Rigid object.
266              */
267             clearForces: function() {
268                 var vec0 = new Ammo.btVector3(0, 0, 0);
269                 this.rigidBody.setLinearVelocity(vec0);
270                 this.rigidBody.setAngularVelocity(vec0);
271                 this.rigidBody.clearForces();
272                 Ammo.destroy(vec0);
273             },
274             /**
275              * 他のRigidとの衝突判定.
276              * @param {enchant.gl.physics.Rigid} rigid 判定するRigid.
277              * @return {Boolean} bool 衝突の有無.
278              */
279             contactTest: function(rigid) {
280                 if (this.world && rigid.world
281                     && this.world == rigid.world) {
282                     return this.world.contactPairTest(this, rigid);
283                 } else {
284                     return false;
285                 }
286             },
287             /**
288              * Rigidを有効化する.
289              * @param {Boolean} force 強制的に有効化する.
290              */
291             activate: function(force) {
292                 this.rigidBody.activate(force);
293             },
294             /**
295              * Rigidに力を加える.
296              * 力はRigidの中心に加えられる.
297              * @param {Number} powerX x軸方向の力.
298              * @param {Number} powerY y軸方向の力.
299              * @param {Number} powerZ z軸方向の力.
300              */
301             applyCentralImpulse: function(powx, powy, powz) {
302                 var powv = new Ammo.btVector3(powx, powy, powz);
303                 this.activate();
304                 this.rigidBody.applyCentralImpulse(powv);
305                 Ammo.destroy(powv);
306             },
307             /**
308              * Rigidに力を加える.
309              * 力は指定した位置に加えられる.
310              * @param {Number} powerX x軸方向の力.
311              * @param {Number} powerY y軸方向の力.
312              * @param {Number} powerZ z軸方向の力.
313              * @param {Number} positonX 力を加える位置のx座標.
314              * @param {Number} positonY 力を加える位置のy座標.
315              * @param {Number} positonZ 力を加える位置のz座標.
316              */
317             applyImpulse: function(powx, powy, powz, posx, posy, posz) {
318                 var powv = new Ammo.btVector3(powx, powy, powz);
319                 var posv = new Ammo.btVector3(posx, posy, posz);
320                 this.activate();
321                 this.rigidBody.applyImpulse(powv, posv);
322                 Ammo.destroy(powv);
323                 Ammo.destroy(posv);
324             },
325             _getTransform: function() {
326                 return this.rigidBody.getWorldTransform();
327             },
328             /**
329              * Rigidをユーザが動かすためのオブジェクトとして設定する.
330              */
331             kinematize: function() {
332                 var flag = this.rigidBody.getCollisionFlags();
333                 this.rigidBody.setCollisionFlags(flag | 2);
334                 this.rigidBody.setActivationState(4);
335             },
336            /**
337             * Rigidの反発係数.
338             * @type Number
339             */
340             restitution: {
341                 get: function() {
342                     return this._restitution;
343                 },
344                 set: function(n) {
345                     this._restitution = n;
346                     this.rigidBody.setRestitution(n);
347                 }
348             },
349            /**
350             * Rigidの摩擦係数.
351             * @type Number
352             */
353             friction: {
354                 get: function() {
355                     return this._friction;
356                 },
357                 set: function(n) {
358                     this._friction = n;
359                     this.rigidBody.setFriction(n);
360                 }
361             }
362         });

493             /**
494              * 枡型の剛体オブジェクト.
495              * @param {Number} scale 枡の中心から枠までの距離.
496              * @param {Number} mass 剛体の質量.

500              * @extends enchant.gl.physics.Rigid
501              */
502             initialize: function(s, mass) {
503                 var shape = new Ammo.btCompoundShape(s);
504                 var addWall = function(sx, sy, sz, px, py, pz) {
505                     var sc = new Ammo.btVector3(sx, sy, sz);
506                     var tr = new Ammo.btTransform();
507                     tr.setIdentity();
508                     var or = new Ammo.btVector3(px, py, pz);
509                     tr.setOrigin(or);

511                     shape.addChildShape(tr, shp);
512                     Ammo.destroy(sc);
513                     Ammo.destroy(or);
514                     Ammo.destroy(tr);
515                 };
516                 addWall(s, s/8, s, 0, s/8 - s, 0);
517                 addWall(s - s/8, s - s/8 - s/8, s/8, s/8, 0, s/8 - s);
518                 addWall(s - s/8, s - s/8 - s/8, s/8, -s/8, 0, s - s/8);
519                 addWall(s/8, s - s/8 - s/8, s - s/8, s/8 - s, 0, -s/8);
520                 addWall(s/8, s - s/8 - s/8, s - s/8, s - s/8, 0, s/8);

529             /**
530              * Worldを持つScene3D.
531              * 時間を進めることで, addChildされたSprite3Dに物理演算が適用される.
532              * @see enchant.gl.physics.World
533              * @constructs
534              * @extends enchant.gl.Scene3D
535              */
536             initialize: function() {
537                 enchant.gl.Scene3D.call(this);
538                 var game = enchant.Game.instance;

540                 this.isPlaying = false;

934                 var up = vec3.create([0, 1, 0]);

936                 var axis = vec3.create();
937                 vec3.cross(norm, up, axis);
938                 var rad = Math.acos(vec3.dot(up, norm) / (vec3.length(up) * vec3.length(norm)));
939                 var q = new Quat(axis[0], axis[1], axis[2], -rad);
940                 var vertices = new Array();
941                 for (i = 0, l = this.mesh.vertices.length; i < l; i+=3) {
942                     var x = this.mesh.vertices[i];
943                     var y = this.mesh.vertices[i+1];
944                     var z = this.mesh.vertices[i+2];
945                     var arr = q.multiplyVec3([x, y, z]);
946                     vertices.push(arr[0] + nx * dist);
947                     vertices.push(arr[1] + ny * dist);
948                     vertices.push(arr[2] + nz * dist);
949                 }
