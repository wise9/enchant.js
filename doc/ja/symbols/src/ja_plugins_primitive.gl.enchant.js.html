562         proto.createTorus = function(r, r2, v, v2) {
563             if (typeof r == 'undefined') {
564                 r = 1.0;
565             } if (typeof r2 == 'undefined') {
566                 r2 = 0.3;
567             } if (typeof v == 'undefined') {
568                 v = 20;
569             } if (typeof v2 == 'undefined') {
570                 v2 = 20;
571             }
572             var ring = [];
573             var norm = [];
574             var rad;
575             var cos;
576             var sin;
577             for (var i = 0; i < v2; i++) {
578                 rad = Math.PI * 2 * i / (v2-1);
579                 cos = Math.cos(rad);
580                 sin = Math.sin(rad);
581                 ring[ring.length] = 0;
582                 ring[ring.length] = sin * r2;
583                 ring[ring.length] = cos * r2 + r - r2;
584                 norm[norm.length] = 0;
585                 norm[norm.length] = sin;
586                 norm[norm.length] = cos;
587             }
588             var vertices = [];
589             var normals = [];
590             var texCoords = [];
591             for (var i = 0; i < v; i++) {
592                 rad = Math.PI * 2 * i / (v-1);
593                 cos = Math.cos(rad);
594                 sin = Math.sin(rad);
595                 for (var j = 0; j < ring.length; j += 3) {
596                     vertices[vertices.length] = ring[j] * cos - ring[j + 2] * sin;
597                     vertices[vertices.length] = ring[j + 1];
598                     vertices[vertices.length] = ring[j + 2] * cos + ring[j] * sin;
599                     normals[normals.length] = norm[j] * cos - ring[j + 2] * sin;
600                     normals[normals.length] = norm[j + 1];
601                     normals[normals.length] = norm[j + 2] * cos + ring[j] + sin;
602                     texCoords[texCoords.length] = 1.0 - i / (v - 1);
603                     texCoords[texCoords.length] = j / (ring.length - 3);
604                 }
605             }
606             var indices = [];
607             var c, c2;
608             for (var i = 0; i < v - 1; i++) {
609                 for (var j = 0; j < v2 - 1; j++) {
610                     c = v2 * i + j;
611                     c2 = v2 * (i + 1) + j;
612                     indices[indices.length] = c;
613                     indices[indices.length] = c + 1;
614                     indices[indices.length] = c2;
615                     indices[indices.length] = c2 + 1;
616                     indices[indices.length] = c2;
617                     indices[indices.length] = c + 1;
618                 }
619             }
620             var mesh = new enchant.gl.Mesh();
621             mesh.vertices = vertices;
622             mesh.normals = normals;
623             mesh.texCoords = texCoords;
624             mesh.indices = indices;
625             mesh.setBaseColor('#ffffff');
626 
627             return mesh;
628         };
