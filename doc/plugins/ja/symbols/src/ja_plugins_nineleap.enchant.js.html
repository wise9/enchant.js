1 /**
  2  * @fileOverview
  3  * nineleap.enchant.js
  4  * @version 0.3.1 (2012/11/15)
  5  * @requires enchant.js v0.4.0 or later
  6  *
  7  * @description
  8  * enchant.js extension for 9leap.net
  9  * 9leap.net 向けの enchant.js 拡張プラグイン。
 10  * core.end の引数にスコアと結果の文字列を渡すことで、ランキングに登録できる。
 11  * (9leapにアップロードした後のみランキング画面にジャンプする)
 12  *
 13  * @usage
 14 
 15  var core = new Core(320, 320);
 16 
 17  core.onload = function(){
 18  // executed after player pushed "START"
 19  // ...
 20  if(some.condition)core.end(score, result);
 21  };
 22 
 23  core.start();
 24  */
 25 
 26 (function() {
 27 
 28     /**
 29      * @type {Object}
 30      */
 31     enchant.nineleap = { assets: ['start.png', 'end.png'] };
 32 
 33     /**
 34      * @scope enchant.nineleap.Core.prototype
 35      */
 36     enchant.nineleap.Core = enchant.Class.create(enchant.Core, {
 37         /**
 38          * start, gameover の画像を表示し、
 39          * 最後にスコアを送信するように拡張された Core クラス
 40          * @param width
 41          * @param height
 42          * @constructs
 43          */
 44         initialize: function(width, height) {
 45             enchant.Core.call(this, width, height);
 46             this.addEventListener('load', function() {
 47                 var core = this;
 48                 this.startScene = new enchant.nineleap.SplashScene();
 49                 this.startScene.image = this.assets['start.png'];
 50                 this.startScene.addEventListener('touchend', function() {
 51                     if (core.started === false) {
 52                         if (core.onstart != null) {
 53                             core.onstart();
 54                         }
 55                         core.started = true;
 56                         coreStart = true;   // deprecated
 57                     }
 58                     if (core.currentScene === this) {
 59                         core.popScene();
 60                     }
 61                     this.removeEventListener('touchend', arguments.callee);
 62                 });
 63                 this.addEventListener('keydown', function() {
 64                     if (this.started === false) {
 65                         if (this.onstart != null) {
 66                             this.onstart();
 67                         }
 68                         this.started = true;
 69                     }
 70                     if (core.currentScene === core.startScene) {
 71                         core.popScene();
 72                     }
 73                     this.removeEventListener('keydown', arguments.callee);
 74                 });
 75                 this.pushScene(this.startScene);
 76 
 77                 this.endScene = new SplashScene();
 78                 this.endScene.image = this.assets['end.png'];
 79             });
 80             this.scoreQueue = false;
 81             this.started = false;
 82             coreStart = false; // deprecated
 83         },
 84 
 85         loadImage: function(src, callback) {
 86             if (callback == null) {
 87                 callback = function() {
 88                 };
 89             }
 90             this.assets[src] = enchant.Surface.load(src);
 91             this.assets[src].addEventListener('load', callback);
 92         },
 93 
 94         start: function() {
 95             var core = this;
 96 
 97             if (this._intervalID) {
 98                 window.clearInterval(this._intervalID);
 99             } else if (this._assets.length) {
100 
101                 if (enchant.Sound.enabledInMobileSafari && !core._touched &&
102                     enchant.ENV.VENDOR_PREFIX === 'webkit' && enchant.ENV.TOUCH_ENABLED) {
103                     console.log("mobile_first");
104                     var scene = new enchant.Scene();
105                     scene.backgroundColor = '#000';
106                     var size = Math.round(core.width / 10);
107                     var sprite = new enchant.Sprite(core.width, size);
108                     sprite.y = (core.height - size) / 2;
109                     sprite.image = new enchant.Surface(core.width, size);
110                     sprite.image.context.fillStyle = '#fff';
111                     sprite.image.context.font = (size - 1) + 'px bold Helvetica,Arial,sans-serif';
112                     var width = sprite.image.context.measureText('Touch to Start').width;
113                     sprite.image.context.fillText('Touch to Start', (core.width - width) / 2, size - 1);
114                     scene.addChild(sprite);
115                     document.addEventListener('touchstart', function() {
116                         core._touched = true;
117                         core.removeScene(scene);
118                         core.start();
119                     }, true);
120                     core.pushScene(scene);
121                     return;
122                 }
123 
124                 var o = {};
125                 var assets = this._assets.filter(function(asset) {
126                     return asset in o ? false : o[asset] = true;
127                 });
128                 var tAssets = (this._twitterAssets != undefined) ? this._twitterAssets : [];
129                 var nAssets = (this._netpriceData != undefined) ? this._netpriceData : [];
130                 var mAssets = (this._memoryAssets != undefined) ? this._memoryAssets : [];
131                 var loaded = 0;
132                 var total = assets.length + tAssets.length + nAssets.length + mAssets.length;
133                 var i, l;
134                 var loadListener = function() {
135                     var e = new enchant.Event('progress');
136                     e.loaded = ++loaded;
137                     e.total = total;
138                     core.dispatchEvent(e);
139                     if (loaded === total) {
140                         core.removeScene(core.loadingScene);
141                         core.dispatchEvent(new enchant.Event('load'));
142                     }
143                 };
144                 for (i = 0, l = assets.length; i < l; i++) {
145                     this.load(assets[i], loadListener);
146                 }
147                 for (i = 0, l = tAssets.length; i < l; i++) {
148                     this.loadImage(tAssets[i], function() {
149                         var e = new enchant.Event('progress');
150                         e.loaded = ++loaded;
151                         e.total = total;
152                         core.dispatchEvent(e);
153                         if (loaded === total) {
154                             core.removeScene(core.loadingScene);
155                             core.dispatchEvent(new enchant.Event('load'));
156                         }
157                     });
158                 }
159                 for (i = 0, l = mAssets.length; i < l; i++) {
160                     this.loadImage(mAssets[i], function() {
161                         var e = new enchant.Event('progress');
162                         e.loaded = ++loaded;
163                         e.total = total;
164                         core.dispatchEvent(e);
165                         if (loaded === total) {
166                             core.removeScene(core.loadingScene);
167                             core.dispatchEvent(new enchant.Event('load'));
168                         }
169                     });
170                 }
171 
172                 for (i = 0, l = nAssets.length; i < l; i++) {
173                     this.loadImage(nAssets[i], function() {
174                         var e = new enchant.Event('progress');
175                         e.loaded = ++loaded;
176                         e.total = total;
177                         core.dispatchEvent(e);
178                         if (loaded === total) {
179                             core.removeScene(core.loadingScene);
180                             core.dispatchEvent(new enchant.Event('load'));
181                         }
182                     });
183                 }
184                 this.pushScene(this.loadingScene);
185             } else {
186                 this.dispatchEvent(new enchant.Event('load'));
187             }
188             this.currentTime = Date.now();
189             this._intervalID = window.setInterval(function() {
190                 core._tick();
191             }, 1000 / this.fps);
192             this.running = true;
193         },
194 
195         end: function(score, result, img) {
196             if (img !== undefined) {
197                 this.endScene.image = img;
198             }
199             this.pushScene(this.endScene);
200             if (location.hostname === 'r.jsgames.jp') {
201                 var submit = function() {
202                     var id = location.pathname.match(/^\/games\/(\d+)/)[1];
203                     location.replace([
204                         'http://9leap.net/games/', id, '/result',
205                         '?score=', encodeURIComponent(score),
206                         '&result=', encodeURIComponent(result)
207                     ].join(''));
208                 };
209                 this.endScene.addEventListener('touchend', submit);
210                 window.setTimeout(submit, 3000);
211             }
212             enchant.Core.instance.end = function() {
213             };
214         }
215     });
216 
217     /**
218      * @scope enchant.nineleap.SplashScene.prototype
219      */
220     enchant.nineleap.SplashScene = enchant.Class.create(enchant.Scene, {
221         /**
222          * @extends enchant.Scene
223          * @constructs
224          * スプラッシュ画像を表示するシーン。
225          */
226         initialize: function() {
227             enchant.Scene.call(this);
228         },
229 
230         /**
231          * 中央に表示する画像
232          * @type {enchant.Surface}
233          */
234         image: {
235             get: function() {
236                 return this._image;
237             },
238             set: function(image) {
239                 this._image = image;
240 
241                 // discard all child nodes
242                 while (this.firstChild) {
243                     this.removeChild(this.firstChild);
244                 }
245 
246                 // generate an Sprite object and put it on center
247                 var sprite = new Sprite(image.width, image.height);
248                 sprite.image = image;
249                 sprite.x = (this.width - image.width) / 2;
250                 sprite.y = (this.height - image.height) / 2;
251                 this.addChild(sprite);
252             }
253         }
254     });
