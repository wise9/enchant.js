 99 /**
100  * グローバルにライブラリのクラスをエクスポートする.
101  *
102  * 引数に何も渡さない場合enchant.jsで定義されたクラス及びプラグインで定義されたクラス
103  * 全てがエクスポートされる. 引数が一つ以上の場合はenchant.jsで定義されたクラスのみ
104  * がデフォルトでエクスポートされ, プラグインのクラスをエクスポートしたい場合は明示的に
105  * プラグインの識別子を引数として渡す必要がある.
106  *
107  * @example
108  *   enchant();     // 全てのクラスがエクスポートされる
109  *   enchant('');   // enchant.js本体のクラスのみがエクスポートされる
110  *   enchant('ui'); // enchant.js本体のクラスとui.enchant.jsのクラスがエクスポートされる
111  *
112  * @param {...String} [modules] エクスポートするモジュール. 複数指定できる.
113  */
114 var enchant = function(modules) {
115     if (modules != null) {
116         if (!(modules instanceof Array)) {
117             modules = Array.prototype.slice.call(arguments);
118         }
119         modules = modules.filter(function(module) {
120             return [module].join();
121         });
122     }
123     (function include(module, prefix) {
124         var submodules = [],
125             i, len;
126         for (var prop in module){
127             if (module.hasOwnProperty(prop)) {
128                 if (typeof module[prop] === 'function') {
129                     window[prop] = module[prop];
130                 } else if (typeof module[prop] === 'object' && Object.getPrototypeOf(module[prop]) === Object.prototype) {
131                     if (modules == null) {
132                         submodules.push(prop);
133                     } else {
134                         i = modules.indexOf(prefix + prop);
135                         if (i !== -1) {
136                             submodules.push(prop);
137                             modules.splice(i, 1);
138                         }
139                     }
140                 }
141             }
142         }
143 
144         for (i = 0, len = submodules.length; i < len; i++) {
145             include(module[submodules[i]], prefix + submodules[i] + '.');
146         }
147     }(enchant, ''));
148 
149     if (modules != null && modules.length) {
150         throw new Error('Cannot load module: ' + modules.join(', '));
151     }
152 };

193 /**
194  * クラスを作成する.
195  *
196  * ほかのクラスを継承したクラスを作成する場合, コンストラクタはデフォルトで
197  * 継承元のクラスのものが使われる. コンストラクタをオーバーライドする場合継承元の
198  * コンストラクタを適用するには明示的に呼び出す必要がある.
199  *
200  * @example
201  *   var Ball = Class.create({ // 何も継承しないクラスを作成する
202  *       initialize: function(radius) { ... }, // メソッド定義
203  *       fall: function() { ... }
204  *   });
205  *
206  *   var Ball = Class.create(Sprite);  // Spriteを継承したクラスを作成する
207  *   var Ball = Class.create(Sprite, { // Spriteを継承したクラスを作成する
208  *       initialize: function(radius) { // コンストラクタを上書きする
209  *          Sprite.call(this, radius*2, radius*2); // 継承元のコンストラクタを適用する
210  *          this.image = game.assets['ball.gif'];
211  *       }
212  *   });
213  *
214  * @param {Function} [superclass] 継承するクラス.
215  * @param {*} [definition] クラス定義.
216  * @static
217  */
218 enchant.Class.create = function(superclass, definition) {
219     if (arguments.length === 0) {
220         return enchant.Class.create(Object, definition);
221     } else if (arguments.length === 1 && typeof arguments[0] !== 'function') {
222         return enchant.Class.create(Object, arguments[0]);
223     }
224 
225     for (var prop in definition){
226         if (definition.hasOwnProperty(prop)) {
227             if (typeof definition[prop] === 'object' && Object.getPrototypeOf(definition[prop]) === Object.prototype) {
228                 if (!('enumerable' in definition[prop])){
229                     definition[prop].enumerable = true;
230                 }
231             } else {
232                 definition[prop] = { value: definition[prop], enumerable: true, writable: true };
233             }
234         }
235     }
236     var Constructor = function() {
237         if (this instanceof Constructor) {
238             Constructor.prototype.initialize.apply(this, arguments);
239         } else {
240             return new Constructor();
241         }
242     };
243     Constructor.prototype = Object.create(superclass.prototype, definition);
244     Constructor.prototype.constructor = Constructor;
245     if (Constructor.prototype.initialize == null) {
246         Constructor.prototype.initialize = function() {
247             superclass.apply(this, arguments);
248         };
249     }
250 
251     return Constructor;
252 };

592 /**
593  * @scope enchant.EventTarget.prototype
594  */
595 enchant.EventTarget = enchant.Class.create({
596     /**
597      * DOM Event風味の独自イベント実装を行ったクラス.
598      * ただしフェーズの概念はなし.
599      * @constructs
600      */
601     initialize: function() {
602         this._listeners = {};
603     },
604     /**
605      * イベントリスナを追加する.
606      * @param {String} type イベントのタイプ.
607      * @param {function(e:enchant.Event)} listener 追加するイベントリスナ.
608      */
609     addEventListener: function(type, listener) {
610         var listeners = this._listeners[type];
611         if (listeners == null) {
612             this._listeners[type] = [listener];
613         } else if (listeners.indexOf(listener) === -1) {
614             listeners.unshift(listener);
615 
616         }
617     },
618     /**
619      * Synonym for addEventListener
620      * @param {String} type Event type.
621      * @param {function(e:enchant.Event)} listener EventListener added.
622      */
623     on: function() {
624         this.addEventListener.apply(this, arguments);
625     },
626     /**
627      * イベントリスナを削除する.
628      * @param {String} type イベントのタイプ.
629      * @param {function(e:enchant.Event)} listener 削除するイベントリスナ.
630      */
631     removeEventListener: function(type, listener) {
632         var listeners = this._listeners[type];
633         if (listeners != null) {
634             var i = listeners.indexOf(listener);
635             if (i !== -1) {
636                 listeners.splice(i, 1);
637             }
638         }
639     },
640     /**
641      * すべてのイベントリスナを削除する.
642      * @param {String} type イベントのタイプ.
643      */
644     clearEventListener: function(type) {
645         if (type != null) {
646             delete this._listeners[type];
647         } else {
648             this._listeners = {};
649         }
650     },
651     /**
652      * イベントを発行する.
653      * @param {enchant.Event} e 発行するイベント.
654      */
655     dispatchEvent: function(e) {
656         e.target = this;
657         e.localX = e.x - this._offsetX;
