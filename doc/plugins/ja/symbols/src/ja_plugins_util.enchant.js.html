 19 /**
 20  * @scope enchant.util.MutableText.prototype
 21  * @type {*}
 22  */
 23 enchant.util.MutableText = enchant.Class.create(enchant.Sprite, {
 24     /**
 25      * ビットマップフォントを用いたラベルクラス
 26      * (参考: draw.text.js http://d.hatena.ne.jp/nakamura001/20110430/1304181043)
 27      * enchant.js 添付素材の font*.png が利用可能。
 28      *
 29      * @usage
 30      *     var text = new MutableText(0, 0);
 31      *     game.text = 'Hello, world!';
 32      *     game.rootScene.addChild(text);
 33      *
 34      * @constructs
 35      * @param posX
 36      * @param posY
 37      * @param width
 38      */
 39     initialize: function(x, y, width) {
 40         enchant.Sprite.call(this, 0, 0);
 41         this.fontSize = 16;
 42         this.widthItemNum = 16;
 43         this.x = x;
 44         this.y = y;
 45         this.text = '';
 46         if (arguments[2]) {
 47             this.row = Math.floor(arguments[2] / this.fontSize);
 48         }
 49     },
 50     /**
 51      * ラベルの内容を書き換える関数
 52      * @param txt
 53      */
 54     setText: function(txt) {
 55         var i, x, y, wNum, charCode, charPos;
 56         this._text = txt;
 57         if (!this.returnLength) {
 58             this.width = Math.min(this.fontSize * this._text.length, enchant.Game.instance.width);
 59         } else {
 60             this.width = Math.min(this.returnLength * this.fontSize, enchant.Game.instance.width);
 61         }
 62         this.height = this.fontSize * (Math.ceil(this._text.length / this.row) || 1);
 63         this.image = new enchant.Surface(this.width, this.height);
 64         this.image.context.clearRect(0, 0, this.width, this.height);
 65         for (i = 0; i < txt.length; i++) {
 66             charCode = txt.charCodeAt(i);
 67             if (charCode >= 32 && charCode <= 127) {
 68                 charPos = charCode - 32;
 69             } else {
 70                 charPos = 0;
 71             }
 72             x = charPos % this.widthItemNum;
 73             y = (charPos / this.widthItemNum) | 0;
 74             this.image.draw(enchant.Game.instance.assets['font0.png'],
 75                 x * this.fontSize, y * this.fontSize, this.fontSize, this.fontSize,
 76                 (i % this.row) * this.fontSize, ((i / this.row) | 0) * this.fontSize, this.fontSize, this.fontSize);
 77         }
 78     },
 79     /**
 80      * ラベルの内容
 81      * @type {String}
 82      */
 83     text: {
 84         get: function() {
 85             return this._text;
 86         },
 87         set: function(txt) {
 88             this.setText(txt);
 89         }
 90     },
 91     /**
 92      * @type {Number}
 93      */
 94     row: {
 95         get: function() {
 96             return this.returnLength || this.width / this.fontSize;
 97         },
 98         set: function(row) {
 99             this.returnLength = row;
100             this.text = this.text;
101         }
102     }
103 });

105 /**
106  * @scope enchant.util.ScoreLabel.prototype
107  * @type {*}
108  */
109 enchant.util.ScoreLabel = enchant.Class.create(enchant.util.MutableText, {
110     /**
111      * スコアを表示するラベル。
112      * 画像フォントクラス (MutableText) を使って表示する。
113      * @constructs
114      * @param x
115      * @param y
116      */
117     initialize: function(x, y) {
118         enchant.util.MutableText.call(this, 0, 0);
119         switch (arguments.length) {
120             case 2:
121                 this.y = y;
122                 this.x = x;
123                 break;
124             case 1:
125                 this.x = x;
126                 break;
127             default:
128                 break;
129         }
130         this._score = 0;
131         this._current = 0;
132         this.easing = 2.5;
133         this.text = this.label = 'SCORE:';
134         this.addEventListener('enterframe', function() {
135             if (this.easing === 0) {
136                 this.text = this.label + (this._current = this._score);
137             } else {
138                 this._current += Math.ceil((this._score - this._current) / this.easing);
139                 this.text = this.label + this._current;
140             }
141         });
142     },
143     /**
144      * スコア
145      * @type {Number}
146      */
147     score: {
148         get: function() {
149             return this._score;
150         },
151         set: function(newscore) {
152             this._score = newscore;
153         }
154     }
155 });

158  * @type {*}
159  * @scope enchant.util.TimeLabel.prototype
160  */
161 enchant.util.TimeLabel = enchant.Class.create(enchant.util.MutableText, {
162     /**
163      * 残り時間などのタイムを表示するラベル
164      * @constructs
165      * @param x
166      * @param y
167      * @param counttype
168      */
169     initialize: function(x, y, counttype) {
170         enchant.util.MutableText.call(this, 0, 0);
171         switch (arguments.length) {
172             case 3:
173             case 2:
174                 this.y = y;
175                 this.x = x;
176                 break;
177             case 1:
178                 this.x = x;
179                 break;
180             default:
181                 break;
182         }
183         this._time = 0;
184         this._count = 1;// この数を毎フレーム每に足して上げ下げを制御する
185         if (counttype === 'countdown') {
186             this._count = -1;
187         }
188         this.text = this.label = 'TIME:';
189         this.addEventListener('enterframe', function() {
190             this._time += this._count;
191             this.text = this.label + (this._time / enchant.Game.instance.fps).toFixed(2);
192         });
193     },
194     /**
195      * 残り時間
196      * @type {Number}
197      */
198     time: {
199         get: function() {
200             return Math.floor(this._time / enchant.Game.instance.fps);
201         },
202         set: function(newtime) {
203             this._time = newtime * enchant.Game.instance.fps;
204         }
205     }
206 });

208 /**
209  * @type {*}
210  * @scope enchant.util.LifeLabel.prototype
211  */
212 enchant.util.LifeLabel = enchant.Class.create(enchant.Group, {
213     /**
214      * ライフを表示する専用のラベル
215      * icon0.png 内のハートの画像を用いる
216      * @constructs
217      * @param x
218      * @param y
219      * @param maxlife
220      */
221     initialize: function(x, y, maxlife) {
222         enchant.Group.call(this);
223         this.x = x || 0;
224         this.y = y || 0;
225         this._maxlife = maxlife || 9;
226         this._life = 0;
227         this.label = new enchant.util.MutableText(0, 0, 80);
228         this.label.text = 'LIFE:';
229         this.addChild(this.label);
230         this.heart = [];
231         for (var i = 0; i < this._maxlife; i++) {
232             this.heart[i] = new enchant.Sprite(16, 16);
233             this.heart[i].image = enchant.Game.instance.assets['icon0.png'];
234             this.heart[i].x = this.label.width + i * 16;
235             this.heart[i].y = -3;
236             this.heart[i].frame = 10;
237             this.addChild(this.heart[i]);
238         }
239     },
240     /**
241      * 残りライフの数
242      * @type {Number}
243      */
244     life: {
245         get: function() {
246             return this._life;
247         },
248         set: function(newlife) {
249             this._life = newlife;
250             if (this._maxlife < newlife) {
251                 this._life = this._maxlife;
252             }
253             for (var i = 0; i < this._maxlife; i++) {
254                 this.heart[i].visible = (i <= newlife - 1);
255             }
256         }
257     }
258 });

260 /**
261  * @scope enchant.util.Bar
262  * @type {*}
263  */
264 enchant.util.Bar = enchant.Class.create(enchant.Sprite, {
265     /**
266      * イージング付きのバークラス
267      * @constructs
268      * @param x
269      * @param y
270      */
271     initialize: function(x, y) {
272         enchant.Sprite.call(this, 1, 16);
273         this.image = new enchant.Surface(1, 16);// Null用
274         this.image.context.fillColor = 'RGB(0, 0, 256)';
275         this.image.context.fillRect(0, 0, 1, 16);
276         this._direction = 'right';
277         this._origin = 0;
278         this._maxvalue = enchant.Game.instance.width;
279         this._lastvalue = 0;
280         this.value = 0;
281         this.easing = 5;
282         switch (arguments.length) {
283             case 2:
284                 this.y = y;
285                 this.x = x;
286                 this._origin = x;
287                 break;
288             case 1:
289                 this.x = x;
290                 this._origin = x;
291                 break;
292             default:
293                 break;
294         }
295         this.addEventListener('enterframe', function() {
296             if (this.value < 0) {
297                 this.value = 0;
298             }
299             this._lastvalue += (this.value - this._lastvalue) / this.easing;
300             if (Math.abs(this._lastvalue - this.value) < 1.3) {
301                 this._lastvalue = this.value;
302             }
303             this.width = (this._lastvalue) | 0;
304             if (this.width > this._maxvalue) {
305                 this.width = this._maxvalue;
306             }
307             if (this._direction === 'left') {
308                 this._x = this._origin - this.width;
309             } else {
310                 this._x = this._origin;
311             }
312             this._updateCoordinate();
313         });
314     },
315     /**
316      * バーの向き ('right' or 'left')
317      * @default 'right'
318      * @type {String}
319      */
320     direction: {
321         get: function() {
322             return this._direction;
323         },
324         set: function(newdirection) {
325             if (newdirection !== 'right' && newdirection !== 'left') {
326                 // ignore
327             } else {
328                 this._direction = newdirection;
329             }
330         }
331     },
332     /**
333      * x 座標
334      * @type {Number}
335      */
336     x: {
337         get: function() {
338             return this._origin;
339         },
340         set: function(x) {
341             this._x = x;
342             this._origin = x;
343             this._updateCoordinate();
344         }
345     },
346     /**
347      * @type {Number}
348      */
349     maxvalue: {
350         get: function() {
351             return this._maxvalue;
352         },
353         set: function(val) {
354             this._maxvalue = val;
355         }
356     }
357 });

359 /**
360  * @scope enchant.util.VirtualMap.prototype
361  */
362 enchant.util.VirtualMap = enchant.Class.create(enchant.Group, {
363     /**
364      * マップライクな Group
365      * addChildで Sprite 等を追加すると、自動的に mx, my プロパティが追加され、
366      * VirtualMap内での座標で Sprite を操作できる
367      *
368      * 使い方
369      * //20 x 20 メッシュの縦横320ピクセルの盤を作り、その上に16 x 16の駒を8つ並べる
370      * var board = new VirtualMap(20, 20);
371      * board.width = 320;
372      * board.height = 320;
373      * for(var i=0; i<8; i++){
374      *     var piece = new Sprite(16, 16);
375      *     piece.image = game.assets['icon0.gif'];
376      *     board.addChild(piece);
377      *     piece.mx = i + 3;
378      *     piece.my = 16;
379      * }
380      * game.rootScene.addChild(board);
381      *
382      * @param meshWidth
383      * @param meshHeight
384      * @constructs
385      */
386     initialize: function(meshWidth, meshHeight) {
387         enchant.Group.call(this);
388         this.meshWidth = meshWidth || 16;
389         this.meshHeight = meshHeight || 16;
390     },
391     /**
392      * VirtualMap にオブジェクトを追加する (自動的にバインドされる)
393      * @param obj
394      */
395     addChild: function(obj) {
396         enchant.Group.prototype.addChild.call(this, obj);
397         this.bind(obj);
398     },
399     /**
400      * VirtualMap にオブジェクトを追加する
401      * reference で指定したオブジェクトより前に追加される (自動的にバインドされる)。
402      * @param obj
403      * @param reference
404      */
405     insertBefore: function(obj, reference) {
406         enchant.Group.prototype.insertBefore.call(this, obj, reference);
407         this.bind(obj);
408     },
409     /**
410      * オブジェクトを VirtualMap にバインドする。
411      * バインドされたオブジェクトはメッシュ座標 mx, my プロパティを持ち、これを操作することで
412      * VirtualMap の中を移動させることができる。
413      * @param obj
414      */
415     bind: function(obj) {
416         Object.defineProperties(obj, {
417             "mx": {
418                 get: function() {
419                     return Math.floor(this.x / this.parentNode.meshWidth);
420                 },
421                 set: function(arg) {
422                     this.x = Math.floor(arg * this.parentNode.meshWidth);
423                 }
424             },
425             "my": {
426                 get: function() {
427                     return Math.floor(this.y / this.parentNode.meshHeight);
428                 },
429                 set: function(arg) {
430                     this.y = Math.floor(arg * this.parentNode.meshWidth);
431                 }
432             }
433         });
434         obj.mx = 0;
435         obj.my = 0;
436     }
437 });
