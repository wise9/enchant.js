 98 /**
 99  * @scope enchant.tl.ActionEventTarget
100  */
101 enchant.tl.ActionEventTarget = enchant.Class.create(enchant.EventTarget, {
102     /**
103      * イベントリスナの実行時にコンテキストを this.target にするよう書き換えた EventTarget
104      * @constructs
105      * @extends enchant.EventTarget
106      */
107     initialize: function() {
108         enchant.EventTarget.apply(this, arguments);
109     },
110     /**
111      * Issue event.
112      * @param {enchant.Event} e Event issued.
113      */
114     dispatchEvent: function(e) {
115         var target;
116         if (this.node) {
117             target = this.node;
118             e.target = target;
119             e.localX = e.x - target._offsetX;
120             e.localY = e.y - target._offsetY;
121         } else {
122             this.node = null;
123         }
124 
125         if (this['on' + e.type] != null) {
126             this['on' + e.type].call(target, e);
127         }
128         var listeners = this._listeners[e.type];
129         if (listeners != null) {
130             listeners = listeners.slice();
131             for (var i = 0, len = listeners.length; i < len; i++) {
132                 listeners[i].call(target, e);
133             }
134         }
135     }
136 });

138 /**
139  * @scope enchant.tl.Action.prototype
140  */
141 enchant.tl.Action = enchant.Class.create(enchant.tl.ActionEventTarget, {
142     /**
143      * アクションクラス。
144      * アクションはタイムラインを構成する単位であり、
145      * 実行したい処理を指定するためのユニットである。
146      * タイムラインに追加されたアクションは順に実行される。
147      *
148      * アクションが開始・終了された時に actionstart, actiontick イベントが発行され、
149      * また1フレーム経過した時には actiontick イベントが発行される。
150      * これらのイベントのリスナとして実行したい処理を指定する。
151      *
152      * time で指定されたフレーム数が経過すると自動的に次のアクションに移行するが、
153      * null が指定されると、タイムラインの next メソッドが呼ばれるまで移行しない。
154      *
155      * @constructs
156      * @param param
157      * @config {integer} [time] アクションが持続するフレーム数。 null が指定されると無限長
158      * @config {function} [onactionstart] アクションが開始される時のイベントリスナ
159      * @config {function} [onactiontick] アクションが1フレーム経過するときのイベントリスナ
160      * @config {function} [onactionend] アクションがが終了する時のイベントリスナ
161      */
162     initialize: function(param) {
163         enchant.tl.ActionEventTarget.call(this);
164         this.time = null;
165         this.frame = 0;
166         for (var key in param) {
167             if (param.hasOwnProperty(key)) {
168                 if (param[key] != null) {
169                     this[key] = param[key];
170                 }
171             }
172         }
173 
174         var action = this;
175 
176         this.timeline = null;
177         this.node = null;
178 
179         this.addEventListener(enchant.Event.ADDED_TO_TIMELINE, function(evt) {
180             action.timeline = evt.timeline;
181             action.node = evt.timeline.node;
182             action.frame = 0;
183         });
184 
185         this.addEventListener(enchant.Event.REMOVED_FROM_TIMELINE, function() {
186             action.timeline = null;
187             action.node = null;
188             action.frame = 0;
189         });
190 
191         this.addEventListener(enchant.Event.ACTION_TICK, function(evt) {
192             action.frame++;
193             if (action.time != null && action.frame >= action.time) {
194                 evt.timeline.next();
195             }
196         });
197 
198     }
199 });

201 /**
202  * @scope enchant.tl.ParallelAction.prototype
203  */
204 enchant.tl.ParallelAction = enchant.Class.create(enchant.tl.Action, {
205     /**
206      * アクションを並列で実行するためのアクション。
207      * 子アクションを複数持つことができる。
208      * @constructs
209      * @extends enchant.tl.Action
210      */
211     initialize: function(param) {
212         enchant.tl.Action.call(this, param);
213         var timeline = this.timeline;
214         var node = this.node;
215         /**
216          * 子アクション
217          */
218         this.actions = [];
219         /**
220          * 実行が終了したアクション
221          */
222         this.endedActions = [];
223         var that = this;
224 
225         this.addEventListener(enchant.Event.ACTION_START, function(evt) {
226             // start するときは同時
227             for (var i = 0, len = that.actions.length; i < len; i++) {
228                 that.actions[i].dispatchEvent(evt);
229             }
230         });
231 
232         this.addEventListener(enchant.Event.ACTION_TICK, function(evt) {
233             var i, len, timeline = {
234                 next: function() {
235                     var action = that.actions[i];
236                     that.actions.splice(i--, 1);
237                     len = that.actions.length;
238                     that.endedActions.push(action);
239 
240                     // イベントを発行
241                     var e = new enchant.Event("actionend");
242                     e.timeline = this;
243                     action.dispatchEvent(e);
244 
245                     e = new enchant.Event("removedfromtimeline");
246                     e.timeline = this;
247                     action.dispatchEvent(e);
248                 }
249             };
250 
251             var e = new enchant.Event("actiontick");
252             e.timeline = timeline;
253             for (i = 0, len = that.actions.length; i < len; i++) {
254                 that.actions[i].dispatchEvent(e);
255             }
256             // 残りアクションが 0 になったら次のアクションへ
257             if (that.actions.length === 0) {
258                 evt.timeline.next();
259             }
260         });
261 
262         this.addEventListener(enchant.Event.ADDED_TO_TIMELINE, function(evt) {
263             for (var i = 0, len = that.actions.length; i < len; i++) {
264                 that.actions[i].dispatchEvent(evt);
265             }
266         });
267 
268         this.addEventListener(enchant.Event.REMOVED_FROM_TIMELINE, function() {
269             // すべて戻す
270             this.actions = this.endedActions;
271             this.endedActions = [];
272         });
273 
274     }
275 });

277 /**
278  * @scope enchant.tl.Tween.prototype
279  */
280 enchant.tl.Tween = enchant.Class.create(enchant.tl.Action, {
281     /**
282      * トゥイーンクラス。
283      * アクションを扱いやすく拡張したクラス。
284      * オブジェクトの特定のプロパティを、なめらかに変更したい時に用いる。
285      *
286      * コンストラクタに渡す設定オブジェクトに、プロパティの目標値を指定すると、
287      * アクションが実行された時に、目標値までなめらかに値を変更するようなアクションを生成する。
288      *
289      * トゥイーンのイージングも、easing プロパティで指定できる。
290      * デフォルトでは enchant.Easing.LINEAR が指定されている。
291      *
292      * @param params
293      * @constructs
294      * @config {time}
295      * @config {easing} [function]
296      */
297     initialize: function(params) {
298         var origin = {};
299         var target = {};
300         enchant.tl.Action.call(this, params);
301 
302         if (this.easing == null) {
303             // linear
304             this.easing = function(t, b, c, d) {
305                 return c * t / d + b;
306             };
307         }
308 
309         var tween = this;
310         this.addEventListener(enchant.Event.ACTION_START, function() {
311             // トゥイーンの対象とならないプロパティ
312             var excepted = ["frame", "time", "callback", "onactiontick", "onactionstart", "onactionend"];
313             for (var prop in params) {
314                 if (params.hasOwnProperty(prop)) {
315                     // 値の代わりに関数が入っていた場合評価した結果を用いる
316                     var target_val;
317                     if (typeof params[prop] === "function") {
318                         target_val = params[prop].call(tween.node);
319                     } else {
320                         target_val = params[prop];
321                     }
322 
323                     if (excepted.indexOf(prop) === -1) {
324                         origin[prop] = tween.node[prop];
325                         target[prop] = target_val;
326                     }
327                 }
328             }
329         });
330 
331         this.addEventListener(enchant.Event.ACTION_TICK, function(evt) {
332             var ratio = tween.easing(tween.frame + 1, 0, 1, tween.time) - tween.easing(tween.frame, 0, 1, tween.time);
333             for (var prop in target) if (target.hasOwnProperty(prop)) {
334                 if (typeof this[prop] === "undefined")continue;
335                 tween.node[prop] += (target[prop] - origin[prop]) * ratio;
336                 if (Math.abs(tween.node[prop]) < 10e-8) tween.node[prop] = 0;
337             }
338         });
339     }
340 });

342 /**
343  * @scope enchant.tl.Timeline.prototype
344  */
345 enchant.tl.Timeline = enchant.Class.create(enchant.EventTarget, {
346     /**
347      * タイムラインクラス。
348      * アクションを管理するためのクラス。
349      * 操作するノードひとつに対して、必ずひとつのタイムラインが対応する。
350      *
351      * tl.enchant.js を読み込むと、Node クラスを継承したすべてのクラス (Group, Scene, Entity, Label, Sprite)の
352      * tl プロパティに、タイムラインクラスのインスタンスが生成される。
353      * タイムラインクラスは、自身に様々なアクションを追加するメソッドを持っており、
354      * これらを使うことで簡潔にアニメーションや様々な操作をすることができる。
355      * @constructs
356      * @param node 操作の対象となるノード
357      */
358     initialize: function(node) {
359         enchant.EventTarget.call(this);
360         this.node = node;
361         this.queue = [];
362         this.paused = false;
363         this.looped = false;
364 
365         this._parallel = null;
366 
367         this.addEventListener(enchant.Event.ENTER_FRAME, this.tick);
368     },
369     /**
370      * キューの先頭にあるアクションを終了し、次のアクションへ移行する。
371      * アクションの中から呼び出されるが、外から呼び出すこともできる。
372      *
373      * アクション実行中に、アクションが終了した場合、
374      * もう一度 tick() 関数が呼ばれるため、1フレームに複数のアクションが処理される場合もある。
375      * ex.
376      *   sprite.tl.then(function A(){ .. }).then(function B(){ .. });
377      * と記述した場合、最初のフレームで A・B の関数どちらも実行される
378      *
379      */
380     next: function() {
381         var e, action = this.queue.shift();
382         e = new enchant.Event("actionend");
383         e.timeline = this;
384         action.dispatchEvent(e);
385 
386         if (this.looped) {
387             e = new enchant.Event("removedfromtimeline");
388             e.timeline = this;
389             action.dispatchEvent(e);
390 
391             action.frame = 0;
392 
393             this.add(action);
394         } else {
395             // イベントを発行して捨てる
396             e = new enchant.Event("removedfromtimeline");
397             e.timeline = this;
398             action.dispatchEvent(e);
399         }
400         this.dispatchEvent(new enchant.Event("enterframe"));
401     },
402     /**
403      * ターゲットの enterframe イベントのリスナとして登録される関数
404      * 1フレーム経過する際に実行する処理が書かれている。
405      * (キューの先頭にあるアクションに対して、actionstart/actiontickイベントを発行する)
406      */
407     tick: function() {
408         if (this.paused){
409             return;
410         }
411         if (this.queue.length > 0) {
412             var action = this.queue[0];
413             if (action.frame === 0) {
414                 var f;
415                 f = new enchant.Event("actionstart");
416                 f.timeline = this;
417                 action.dispatchEvent(f);
418             }
419             var e = new enchant.Event("actiontick");
420             e.timeline = this;
421             action.dispatchEvent(e);
422         }
423     },
424     add: function(action) {
425         if (this._parallel) {
426             this._parallel.actions.push(action);
427             this._parallel = null;
428         } else {
429             this.queue.push(action);
430         }
431         action.frame = 0;
432 
433         var e = new enchant.Event("addedtotimeline");
434         e.timeline = this;
435         action.dispatchEvent(e);
436 
437         e = new enchant.Event("actionadded");
438         e.action = action;
439         this.dispatchEvent(e);
440 
441         return this;
442     },
443     /**
444      * アクションを簡単に追加するためのメソッド。
445      * 実体は add メソッドのラッパ。
446      * @param params アクションの設定オブジェクト
447      */
448     action: function(params) {
449         return this.add(new enchant.tl.Action(params));
450     },
451     /**
452      * トゥイーンを簡単に追加するためのメソッド。
453      * 実体は add メソッドのラッパ。
454      * @param params トゥイーンの設定オブジェクト。
455      */
456     tween: function(params) {
457         return this.add(new enchant.tl.Tween(params));
458     },
459     /**
460      * タイムラインのキューをすべて破棄する。終了イベントは発行されない。
461      */
462     clear: function() {
463         var e = new enchant.Event("removedfromtimeline");
464         e.timeline = this;
465 
466         for (var i = 0, len = this.queue.length; i < len; i++) {
467             this.queue[i].dispatchEvent(e);
468         }
469         this.queue = [];
470         return this;
471     },
472     /**
473      * タイムラインを早送りする。
474      * 指定したフレーム数が経過したのと同様の処理を、瞬時に実行する。
475      * 巻き戻しはできない。
476      * @param frames
477      */
478     skip: function(frames) {
479         while (frames--) {
480             this.dispatchEvent(new enchant.Event("enterframe"));
481         }
482         return this;
483     },
484     /**
485      * タイムラインの実行を一時停止する
486      */
487     pause: function() {
488         this.paused = true;
489         return this;
490     },
491     /**
492      * タイムラインの実行を再開する
493      */
494     resume: function() {
495         this.paused = false;
496         return this;
497     },
498     /**
499      * タイムラインをループさせる。
500      * ループしているときに終了したアクションは、タイムラインから取り除かれた後
501      * 再度タイムラインに追加される。このアクションは、ループが解除されても残る。
502      */
503     loop: function() {
504         this.looped = true;
505         return this;
506     },
507     /**
508      * タイムラインのループを解除する。
509      */
510     unloop: function() {
511         this.looped = false;
512         return this;
513     },
514     /**
515      * 指定したフレーム数だけ待ち、何もしないアクションを追加する。
516      * @param time
517      */
518     delay: function(time) {
519         this.add(new enchant.tl.Action({
520             time: time
521         }));
522         return this;
523     },
524     /**
525      * @ignore
526      * @param time
527      */
528     wait: function(time) {
529         // reserved
530         return this;
531     },
532     /**
533      * 関数を実行し、即時に次のアクションに移るアクションを追加する。
534      * @param func
535      */
536     then: function(func) {
537         var timeline = this;
538         this.add(new enchant.tl.Action({
539             onactiontick: function(evt) {
540                 func.call(timeline.node);
541                 timeline.next();
542             }
543         }));
544         return this;
545     },
546     /**
547      * then メソッドのシノニム。
548      * 関数を実行し、即時に次のアクションに移る。
549      * @param func
550      */
551     exec: function(func) {
552         this.then(func);
553     },
554     /**
555      * 実行したい関数を、フレーム数をキーとした連想配列(オブジェクト)で複数指定し追加する。
556      * 内部的には delay, then を用いている。
557      *
558      * @example
559      * sprite.tl.cue({
560      *    10: function(){ 10フレーム経過した後に実行される関数 },
561      *    20: function(){ 20フレーム経過した後に実行される関数 },
562      *    30: function(){ 30フレーム経過した後に実行される関数 }
563      * });
564      * @param cue キューオブジェクト
565      */
566     cue: function(cue) {
567         var ptr = 0;
568         for (var frame in cue) {
569             if (cue.hasOwnProperty(frame)) {
570                 this.delay(frame - ptr);
571                 this.then(cue[frame]);
572                 ptr = frame;
573             }
574         }
575     },
576     /**
577      * ある関数を指定したフレーム数繰り返し実行するアクションを追加する。
578      * @param func 実行したい関数
579      * @param time 持続フレーム数
580      */
581     repeat: function(func, time) {
582         this.add(new enchant.tl.Action({
583             onactiontick: function(evt) {
584                 func.call(this);
585             },
586             time: time
587         }));
588         return this;
589     },
590     /**
591      * 複数のアクションを並列で実行したいときに指定する。
592      * and で結ばれたすべてのアクションが終了するまで次のアクションには移行しない
593      * @example
594      * sprite.tl.fadeIn(30).and.rotateBy(360, 30);
595      * 30フレームでフェードインしながら 360度回転する
596      */
597     and: function() {
598         var last = this.queue.pop();
599         if (last instanceof enchant.tl.ParallelAction) {
600             this._parallel = last;
601             this.queue.push(last);
602         } else {
603             var parallel = new enchant.tl.ParallelAction();
604             parallel.actions.push(last);
605             this.queue.push(parallel);
606             this._parallel = parallel;
607         }
608         return this;
609     },
610     /**
611      * @ignore
612      */
613     or: function() {
614         return this;
615     },
616     /**
617      * @ignore
618      */
619     doAll: function(children) {
620         return this;
621     },
622     /**
623      * @ignore
624      */
625     waitAll: function() {
626         return this;
627     },
628     /**
629      * true値 が返るまで、関数を毎フレーム実行するアクションを追加する。
630      * @example
631      * sprite.tl.waitUntil(function(){
632      *    return this.x-- < 0
633      * }).then(function(){ .. });
634      * // x 座標が負になるまで毎フレーム x座標を減算し続ける
635      *
636      * @param func 実行したい関数
637      */
638     waitUntil: function(func) {
639         var timeline = this;
640         this.add(new enchant.tl.Action({
641             onactionstart: func,
642             onactiontick: function(evt) {
643                 if (func.call(this)) {
644                     timeline.next();
645                 }
646             }
647         }));
648         return this;
649     },
650     /**
651      * Entity の不透明度をなめらかに変えるアクションを追加する。
652      * @param opacity 目標の不透明度
653      * @param time フレーム数

864 /**
865  * イージング関数ライブラリ
866  * ActionScript で広く使われている
867  * Robert Penner による Easing Equations を JavaScript に移植した。
868  * @type {Object}
869  */
870 enchant.Easing = {
871     LINEAR: function(t, b, c, d) {
872         return c * t / d + b;
873     },
874     SWING: function(t, b, c, d) {
875         return c * (0.5 - Math.cos(((t / d)*Math.PI))/2)+ b;
876     },
877     // quad
878     QUAD_EASEIN: function(t, b, c, d) {
879         return c * (t /= d) * t + b;
880     },
881     QUAD_EASEOUT: function(t, b, c, d) {
882         return -c * (t /= d) * (t - 2) + b;
883     },
884     QUAD_EASEINOUT: function(t, b, c, d) {
885         if ((t /= d / 2) < 1) {
886             return c / 2 * t * t + b;
887         }
888         return -c / 2 * ((--t) * (t - 2) - 1) + b;
889     },
890     // cubic
891     CUBIC_EASEIN: function(t, b, c, d) {
892         return c * (t /= d) * t * t + b;
893     },
894     CUBIC_EASEOUT: function(t, b, c, d) {
895         return c * ((t = t / d - 1) * t * t + 1) + b;
896     },
897     CUBIC_EASEINOUT: function(t, b, c, d) {
898         if ((t /= d / 2) < 1) {
899             return c / 2 * t * t * t + b;
900         }
901         return c / 2 * ((t -= 2) * t * t + 2) + b;
902     },
903     // quart
904     QUART_EASEIN: function(t, b, c, d) {
905         return c * (t /= d) * t * t * t + b;
906     },
907     QUART_EASEOUT: function(t, b, c, d) {
908         return -c * ((t = t / d - 1) * t * t * t - 1) + b;
909     },
910     QUART_EASEINOUT: function(t, b, c, d) {
911         if ((t /= d / 2) < 1) {
912             return c / 2 * t * t * t * t + b;
913         }
914         return -c / 2 * ((t -= 2) * t * t * t - 2) + b;
915     },
916     // quint
917     QUINT_EASEIN: function(t, b, c, d) {
918         return c * (t /= d) * t * t * t * t + b;
919     },
920     QUINT_EASEOUT: function(t, b, c, d) {
921         return c * ((t = t / d - 1) * t * t * t * t + 1) + b;
922     },
923     QUINT_EASEINOUT: function(t, b, c, d) {
924         if ((t /= d / 2) < 1) {
925             return c / 2 * t * t * t * t * t + b;
926         }
927         return c / 2 * ((t -= 2) * t * t * t * t + 2) + b;
928     },
929     //sin
930     SIN_EASEIN: function(t, b, c, d) {
931         return -c * Math.cos(t / d * (Math.PI / 2)) + c + b;
932     },
933     SIN_EASEOUT: function(t, b, c, d) {
934         return c * Math.sin(t / d * (Math.PI / 2)) + b;
935     },
936     SIN_EASEINOUT: function(t, b, c, d) {
937         return -c / 2 * (Math.cos(Math.PI * t / d) - 1) + b;
938     },
939     // circ
940     CIRC_EASEIN: function(t, b, c, d) {
941         return -c * (Math.sqrt(1 - (t /= d) * t) - 1) + b;
942     },
943     CIRC_EASEOUT: function(t, b, c, d) {
944         return c * Math.sqrt(1 - (t = t / d - 1) * t) + b;
945     },
946     CIRC_EASEINOUT: function(t, b, c, d) {
947         if ((t /= d / 2) < 1) {
948             return -c / 2 * (Math.sqrt(1 - t * t) - 1) + b;
949         }
950         return c / 2 * (Math.sqrt(1 - (t -= 2) * t) + 1) + b;
951     },
952     // elastic
953     ELASTIC_EASEIN: function(t, b, c, d, a, p) {
954         if (t === 0) {
955             return b;
956         }
957         if ((t /= d) === 1) {
958             return b + c;
959         }
960 
961         if (!p) {
962             p = d * 0.3;
963         }
964 
965         var s;
966         if (!a || a < Math.abs(c)) {
967             a = c;
968             s = p / 4;
969         } else {
970             s = p / (2 * Math.PI) * Math.asin(c / a);
971         }
972         return -(a * Math.pow(2, 10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p)) + b;
973     },
974     ELASTIC_EASEOUT: function(t, b, c, d, a, p) {
975         if (t === 0) {
976             return b;
977         }
978         if ((t /= d) === 1) {
979             return b + c;
980         }
981         if (!p) {
982             p = d * 0.3;
983         }
984         var s;
985         if (!a || a < Math.abs(c)) {
986             a = c;
987             s = p / 4;
988         } else {
989             s = p / (2 * Math.PI) * Math.asin(c / a);
990         }
991         return (a * Math.pow(2, -10 * t) * Math.sin((t * d - s) * (2 * Math.PI) / p) + c + b);
992     },
993     ELASTIC_EASEINOUT: function(t, b, c, d, a, p) {
994         if (t === 0) {
995             return b;
996         }
997         if ((t /= d / 2) === 2) {
998             return b + c;
999         }
1000         if (!p) {
1001             p = d * (0.3 * 1.5);
1002         }
1003         var s;
1004         if (!a || a < Math.abs(c)) {
1005             a = c;
1006             s = p / 4;
1007         } else {
1008             s = p / (2 * Math.PI) * Math.asin(c / a);
1009         }
1010         if (t < 1) {
1011             return -0.5 * (a * Math.pow(2, 10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p)) + b;
1012         }
1013         return a * Math.pow(2, -10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p) * 0.5 + c + b;
1014     },
1015     // bounce
1016     BOUNCE_EASEOUT: function(t, b, c, d) {
1017         if ((t /= d) < (1 / 2.75)) {
1018             return c * (7.5625 * t * t) + b;
1019         } else if (t < (2 / 2.75)) {
1020             return c * (7.5625 * (t -= (1.5 / 2.75)) * t + 0.75) + b;
1021         } else if (t < (2.5 / 2.75)) {
1022             return c * (7.5625 * (t -= (2.25 / 2.75)) * t + 0.9375) + b;
1023         } else {
1024             return c * (7.5625 * (t -= (2.625 / 2.75)) * t + 0.984375) + b;
1025         }
1026     },
1027     BOUNCE_EASEIN: function(t, b, c, d) {
1028         return c - enchant.Easing.BOUNCE_EASEOUT(d - t, 0, c, d) + b;
1029     },
1030     BOUNCE_EASEINOUT: function(t, b, c, d) {
1031         if (t < d / 2) {
1032             return enchant.Easing.BOUNCE_EASEIN(t * 2, 0, c, d) * 0.5 + b;
1033         } else {
1034             return enchant.Easing.BOUNCE_EASEOUT(t * 2 - d, 0, c, d) * 0.5 + c * 0.5 + b;
1035         }
1036 
1037     },
1038     // back
1039     BACK_EASEIN: function(t, b, c, d, s) {
1040         if (s === undefined) {
1041             s = 1.70158;
1042         }
1043         return c * (t /= d) * t * ((s + 1) * t - s) + b;
1044     },
1045     BACK_EASEOUT: function(t, b, c, d, s) {
1046         if (s === undefined) {
1047             s = 1.70158;
1048         }
1049         return c * ((t = t / d - 1) * t * ((s + 1) * t + s) + 1) + b;
1050     },
1051     BACK_EASEINOUT: function(t, b, c, d, s) {
1052         if (s === undefined) {
1053             s = 1.70158;
1054         }
1055         if ((t /= d / 2) < 1) {
1056             return c / 2 * (t * t * (((s *= (1.525)) + 1) * t - s)) + b;
1057         }
1058         return c / 2 * ((t -= 2) * t * (((s *= (1.525)) + 1) * t + s) + 2) + b;
1059     },
1060     // expo
1061     EXPO_EASEIN: function(t, b, c, d) {
1062         return (t === 0) ? b : c * Math.pow(2, 10 * (t / d - 1)) + b;
1063     },
1064     EXPO_EASEOUT: function(t, b, c, d) {
1065         return (t === d) ? b + c : c * (-Math.pow(2, -10 * t / d) + 1) + b;
1066     },
1067     EXPO_EASEINOUT: function(t, b, c, d) {
1068         if (t === 0) {
1069             return b;
1070         }
1071         if (t === d) {
1072             return b + c;
1073         }
1074         if ((t /= d / 2) < 1) {
1075             return c / 2 * Math.pow(2, 10 * (t - 1)) + b;
1076         }
1077         return c / 2 * (-Math.pow(2, -10 * --t) + 2) + b;
1078     }
