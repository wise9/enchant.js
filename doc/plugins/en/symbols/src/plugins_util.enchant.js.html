 24 // 画像でフォントを再現したラベル (参考: draw.text.js)
 25 enchant.util.MutableText = enchant.Class.create(enchant.Sprite, {
 26     initialize: function(posX, posY, width) {
 27         enchant.Sprite.call(this, 0, 0);
 28         this.fontSize = 16;
 29         this.widthItemNum = 16;
 30         // font.png の横の文字数
 31         this.x = posX;
 32         this.y = posY;
 33         this.text = '';
 34         if (arguments[2]) {
 35             this.row = Math.floor(arguments[2] / this.fontSize);
 36         }
 37     },
 38     setText: function(txt) {
 39         var i, x, y, wNum, charCode, charPos;
 40         this._text = txt;
 41         if (!this.returnLength) {
 42             this.width = Math.min(this.fontSize * this._text.length, enchant.Game.instance.width);
 43         } else {
 44             this.width = Math.min(this.returnLength * this.fontSize, enchant.Game.instance.width);
 45         }
 46         this.height = this.fontSize * (Math.ceil(this._text.length / this.row) || 1);
 47         this.image = new Surface(this.width, this.height);
 48         this.image.context.clearRect(0, 0, this.width, this.height);
 49         for (i = 0; i < txt.length; i++) {
 50             charCode = txt.charCodeAt(i);
 51             if (charCode >= 32 && charCode <= 127) {
 52                 charPos = charCode - 32;
 53             } else {
 54                 charPos = 0;
 55             }
 56             x = charPos % this.widthItemNum;
 57             y = (charPos / this.widthItemNum) | 0;
 58             this.image.draw(enchant.Game.instance.assets['font.png'],
 59                 x * this.fontSize, y * this.fontSize, this.fontSize, this.fontSize,
 60                 (i % this.row) * this.fontSize, ((i / this.row) | 0) * this.fontSize, this.fontSize, this.fontSize);
 61         }
 62     },
 63     text: {
 64         get: function() {
 65             return this._text;
 66         },
 67         set: function(txt) {
 68             this.setText(txt);
 69         }
 70     },
 71     row: {
 72         get: function() {
 73             return this.returnLength || this.width / this.fontSize;
 74         },
 75         set: function(row) {
 76             this.returnLength = row;
 77             this.text = this.text;
 78         }
 79     }
 80 });

 82 // スコアラベル
 83 enchant.util.ScoreLabel = enchant.Class.create(enchant.util.MutableText, {
 84     initialize: function(x, y) {
 85         MutableText.call(this, 0, 0);
 86         switch (arguments.length) {
 87             case 2:
 88                 this.y = y;
 89             case 1:
 90                 this.x = x;
 91                 break;
 92             default:
 93                 break;
 94         }
 95         this._score = 0;
 96         this._current = 0;
 97         this.easing = 2.5;
 98         this.text = this.label = 'SCORE:';
 99         this.addEventListener('enterframe', function() {
100             if (this.easing == 0) {
101                 this.text = this.label + (this._current = this._score);
102             } else {
103                 this._current += Math.ceil((this._score - this._current) / this.easing);
104                 this.text = this.label + this._current;
105             }
106         });
107     },
108     score: {
109         get: function() {
110             return this._score;
111         },
112         set: function(newscore) {
113             this._score = newscore;
114         }
115     }
116 });

118 // タイムラベル
119 enchant.util.TimeLabel = enchant.Class.create(enchant.util.MutableText, {
120     initialize: function(x, y, counttype) {
121         MutableText.call(this, 0, 0);
122         switch (arguments.length) {
123             case 3:
124             case 2:
125                 this.y = y;
126             case 1:
127                 this.x = x;
128                 break;
129             default:
130                 break;
131         }
132         this._time = 0;
133         this._count = 1;// この数を毎フレーム每に足して上げ下げを制御する
134         if (counttype == 'countdown')this._count = -1;
135         this.text = this.label = 'TIME:';
136         this.addEventListener('enterframe', function() {
137             this._time += this._count;
138             this.text = this.label + (this._time / enchant.Game.instance.fps).toFixed(2);
139         });
140     },
141     time: {
142         get: function() {
143             return Math.floor(this._time / enchant.Game.instance.fps);
144         },
145         set: function(newtime) {
146             this._time = newtime * enchant.Game.instance.fps;
147         }
148     }
149 });

151 // ライフラベル
152 enchant.util.LifeLabel = enchant.Class.create(enchant.Group, {
153     initialize: function(x, y, maxlife) {
154         Group.call(this);
155         this.x = x || 0;
156         this.y = y || 0;
157         this._maxlife = maxlife || 9;
158         this._life = 0;
159         this.label = new MutableText(0, 0, 80, 'LIFE:')
160         this.addChild(this.label);
161         this.heart = [];
162         for (var i = 0; i < this._maxlife; i++) {
163             this.heart[i] = new Sprite(16, 16);
164             this.heart[i].image = enchant.Game.instance.assets['icon0.gif'];
165             this.heart[i].x = this.label.width + i * 16;
166             this.heart[i].y = -3;
167             this.heart[i].frame = 11;
168             this.addChild(this.heart[i]);
169         }
170     },
171     life: {
172         get: function() {
173             return this._life;
174         },
175         set: function(newlife) {
176             this._life = newlife;
177             if (this._maxlife < newlife) {
178                 this._life = this._maxlife;
179             }
180             for (var i = 0; i < this._maxlife; i++) {
181                 if (i <= newlife - 1) {
182                     this.heart[i].visible = true;
183                 } else {
184                     this.heart[i].visible = false;
185                 }
186             }
187         }
188     }
189 });

191 // イージング付きのバー (左右方向のみ) 
192 enchant.Bar = enchant.Class.create(enchant.Sprite, {
193     initialize: function(x, y) {
194         Sprite.call(this, 1, 16);
195         this.image = new Surface(1, 16);// Null用
196         this.image.context.fillColor = 'RGB(0, 0, 256)';
197         this.image.context.fillRect(0, 0, 1, 16);
198         this._direction = 'right';
199         this._origin = 0;
200         this._maxvalue = enchant.Game.instance.width;
201         this._lastvalue = 0;
202         this.value = 0;
203         this.easing = 5;
204         switch (arguments.length) {
205             case 2:
206                 this.y = y;
207             case 1:
208                 this.x = x;
209                 this._origin = x;
210             default:
211                 break;
212         }
213         this.addEventListener('enterframe', function() {
214             if (this.value < 0)this.value = 0;
215             this._lastvalue += (this.value - this._lastvalue) / this.easing;
216             if (Math.abs(this._lastvalue - this.value) < 1.3)this._lastvalue = this.value;
217             this.width = (this._lastvalue) | 0;
218             if (this.width > this._maxvalue)this.width = this._maxvalue;
219             if (this._direction == 'left') {
220                 this._x = this._origin - this.width;
221             } else {
222                 this._x = this._origin;
223             }
224             this._updateCoordinate();
225         });
226     },
227     direction: {
228         get: function() {
229             return this._direction;
230         },
231         set: function(newdirection) {
232             if (newdirection != 'right' && newdirection != 'left') {
233                 console.warn(newdirection + ' は未定義の向きです。rightかleftを指定してください.');
234             } else {
235                 this._direction = newdirection;
236             }
237         }
238     },
239     x: {
240         get: function() {
241             return this._origin;
242         },
243         set: function(x) {
244             this._x = x;
245             this._origin = x;
246             this._updateCoordinate();
247         }
248     },
249     maxvalue: {
250         get: function() {
251             return this._maxvalue;
252         },
253         set: function(val) {
254             this._maxvalue = val;
255         }
256     }
257 });

259 /**
260  * マップライクな Group
261  * addChildで Sprite 等を追加すると、自動的に mx, my プロパティが追加され、
262  * VirtualMap内での座標で Sprite を操作できる
263  *
264  * 使い方
265  * //20 x 20 メッシュの縦横320ピクセルの盤を作り、その上に16 x 16の駒を8つ並べる
266  * var board = new VirtualMap(20, 20);
267  * board.width = 320;
268  * board.height = 320;
269  * for(var i=0; i<8; i++){
270  *     var piece = new Sprite(16, 16);
271  *     piece.image = game.assets['icon0.gif'];
272  *     board.addChild(piece);
273  *     piece.mx = i + 3;
274  *     piece.my = 16;
275  * }
276  * game.rootScene.addChild(board);
277  */
278 enchant.util.VirtualMap = enchant.Class.create(enchant.Group, {
279     initialize: function(meshWidth, meshHeight) {
280         enchant.Group.call(this);
281         this.meshWidth = meshWidth || 16;
282         this.meshHeight = meshHeight || 16;
283     },
284     addChild: function(obj) {
285         enchant.Group.prototype.addChild.call(this, obj);
286         this.bind(obj);
287     },
288     insertBefore: function(obj, reference) {
289         enchant.Group.prototype.insertBefore.call(this, obj, reference);
290         this.bind(obj);
291     },
292     bind: function(obj) {
293         Object.defineProperties(obj, {
294             "mx": {
295                 get: function() {
296                     return Math.floor(this.x / this.parentNode.meshWidth);
297                 },
298                 set: function(arg) {
299                     this.x = Math.floor(arg * this.parentNode.meshWidth);
300                 }
301             },
302             "my": {
303                 get: function() {
304                     return Math.floor(this.y / this.parentNode.meshHeight);
305                 },
306                 set: function(arg) {
307                     this.y = Math.floor(arg * this.parentNode.meshWidth);
308                 }
309             }
310         });
311         obj.mx = 0;
312         obj.my = 0;
313     }
314 });
