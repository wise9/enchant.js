173     var getElementMetrics = function(string, font) {
174         var e = document.createElement('div');
175         var cvs = document.createElement('canvas');
176         var ctx = cvs.getContext('2d');
177         var arr, str, w;
178         var width = 0;
179         var height = 0;
180         if (!font) {
181             font = enchant.widget._env.font;
182         }
183         ctx.font = font;
184         e.style.font = font;
185         string = string || '';
186         string = string.replace(/<(br|BR) ?\/?>/g, '<br>');
187         arr = string.split('<br>');
188         for (var i = 0, l = arr.length; i < l; i++) {
189             str = arr[i];
190             w = ctx.measureText(str).width;
191             if (width < w) {
192                 width = w;
193             }
194         }
195 
196         e.innerHTML = string;
197 
198         if (document.body) {
199             document.body.appendChild(e);
200             height = parseInt(getComputedStyle(e).height, 10);
201             e.style.position = 'absolute';
202             width = parseInt(getComputedStyle(e).width, 10);
203             document.body.removeChild(e);
204         } else {
205             height = 14 * arr.length;
206         }
207 
208         return {
209             width: width + 1,
210             height: height + 1
211         };
212     };

234     var distribute = function(value, div) {
235         if (typeof div == 'array') {
236             var ratio = div;
237             var ret = new Array(ratio.length);
238             var retSum = 0;
239             var maxi = 0;
240             var max = 0;
241             var sum = 0;
242             var quo;
243 
244             ratio.forEach(function(n) {
245                 sum += n;
246             });
247             quo = value / sum;
248 
249             for (var i = 0, l = ret.length; i < l; i++) {
250                 ret[i] = Math.round(quo * ratio[i]);
251                 if (ratio[i] < max) {
252                     maxi = i;
253                     max = ratio[i];
254                 }
255             }
256 
257             ret.forEach(function(n) {
258                 retSum += n;
259             });
260 
261             ret[maxi] += value - retSum;
262         } else if (typeof div == 'number') {
263             var ret = new Array(div);
264             var quo = ~~(value / div);
265             var rem = ~~(value % div);
266             for (var i = 0, l = div; i < l; i++) {
267                 ret[i] = quo;
268             }
269             for (var i = 0, l = rem; i < l; i++) {
270                 ret[i % div] += 1;
271             }
272         }
273         return ret;
274     };

517     /**
518      * Perform pushScene with transition animation.
519      * @param {enchant.Scene} inScene New scene transitioned to.
520      * @return {enchant.Scene} New scene
521      * @requires widget.enchant.js
522      */
523     enchant.Core.prototype.transitionPush = function(inScene) {
524         if (_transitionLock) return null;
525         _transitionLock = true;
526         var time = 15;
527         var c = 0;
528         var outScene = this.currentScene;
529         Effect.transitForwardIn.call(inScene, time);
530         Effect.transitForwardOut.call(outScene, time);
531         this.addEventListener(enchant.Event.ENTER_FRAME, function(e) {
532             outScene.dispatchEvent(e);
533             if (c > time) {
534                 _transitionLock = false;
535                 this.removeEventListener(enchant.Event.ENTER_FRAME, arguments.callee);
536                 inScene.dispatchEvent(new enchant.Event(enchant.Event.TRANSITIONENTER));
537                 outScene.dispatchEvent(new enchant.Event(enchant.Event.TRANSITIONEXIT));
538             }
539             c++;
540         });
541         return this.pushScene(inScene);
542     };

544     /**
545      * Perform popScene with transition animation.
546      * @return {enchant.Scene} Finished scene.
547      * @requires widget.enchant.js
548      */
549     enchant.Core.prototype.transitionPop = function() {
550         if (_transitionLock) return null;
551         if (this.currentScene == this.rootScene) return null;
552         _transitionLock = true;
553         var time = 15;
554         var c = 0;
555         var outScene = this.currentScene;
556         var inScene = this._scenes[this._scenes.length - 2];
557         this.addEventListener(enchant.Event.ENTER_FRAME, function(e) {
558             inScene.dispatchEvent(e);
559             if (c > time) {
560                 _transitionLock = false;
561                 this.removeEventListener(enchant.Event.ENTER_FRAME, arguments.callee);
562                 this.popScene();
563                 outScene.dispatchEvent(new enchant.Event(enchant.Event.TRANSITIONEXIT));
564                 inScene.dispatchEvent(new enchant.Event(enchant.Event.TRANSITIONENTER));
565             }
566             c++;
567         });
568         Effect.transitBackIn.call(inScene, time);
569         Effect.transitBackOut.call(outScene, time);
570         return this._scenes[this._scenes.length - 1];
571     };

576     enchant.widget.GestureDetector = enchant.Class.create(enchant.EventTarget, {
577         /**
578          * Issue event after detecting several gestures.
579          * Can detect tap, double tap, hold, drag, flick, and more.
580          * @param {enchant.Entity} target Object for which you wish to detect input.
581          * @constructs
582          * @extends enchant.EventTarget
583          */
584         initialize: function(target) {
585             var core = enchant.Core.instance;
586             enchant.EventTarget.call(this);
587             this._target;
588             this._startX = 0;
589             this._startY = 0;
590             this._lastX = 0;
591             this._lastY = 0;
592             this._touchElapsed = 0;
593             this._releaseElapsed = 0;
594             this._state = NOTOUCH;
595             this._velobase = (core.width > core.height) ? core.height : core.width;
596 
597             var detector = this;
598             this._handler = function(e) {
599                 detector.dispatchEvent(e);
600             };
601 
602             this._types = [
603                 enchant.Event.TOUCH_START,
604                 enchant.Event.TOUCH_MOVE,
605                 enchant.Event.TOUCH_END,
606                 enchant.Event.ENTER_FRAME
607             ];
608 
609             if (target) {
610                 this.attach(target);
611             }
612         },
613         attach: function(target) {
614             this._target = target;
615             this._types.forEach(function(event) {
616                 this._target.addEventListener(event, this._handler);
617             }, this);
618         },
619         detach: function() {
620             this._types.forEach(function(event) {
621                 this._target.removeEventListener(event, this._handler);
622             }, this);
623             this._target = null;
624         },
625         ontouchstart: function(e) {
626             var core = enchant.Core.instance;
627             this._startFrame = core.frame;
628             this._startX = this._lastX = e.x;
629             this._startY = this._lastY = e.y;
630             if (this._state == WAITDBL) {
631                 this._state = NOMOVEDBL;
632             } else if (this._state == NOTOUCH) {
633                 this._state = NOMOVE;
634             }
635         },
636         ontouchmove: function(e) {
637             var dx = e.x - this._lastX;
638             var dy = e.y - this._lastY;
639             this._lastX = e.x;
640             this._lastY = e.y;
641             switch (this._state) {
642                 case NOMOVE:
643                 case NOMOVEDBL:
644                     this._state = MOVED;
645                 case MOVED:
646                     var evt = new enchant.Event(enchant.Event.SLIP);
647                     evt.x = this._lastX;
648                     evt.y = this._lastY;
649                     evt.dx = dx;
650                     evt.dy = dy;
651                     this._target.dispatchEvent(evt);
652                     break;
653                 case HOLD:
654                     var evt = new enchant.Event(enchant.Event.DRAG);
655                     evt.x = this._lastX;
656                     evt.y = this._lastY;
657                     evt.dx = dx;
658                     evt.dy = dy;
659                     this._target.dispatchEvent(evt);
660                     break;
661                 default:
662                     break;
663             }
664         },
665         ontouchend: function(e) {
666             var core = enchant.Core.instance;
667             switch (this._state) {
668                 case MOVED:
669                     velocityX = (this._lastX - this._startX) / this._velobase / this._touchElapsed * 1000;
670                     velocityY = (this._lastY - this._startY) / this._velobase / this._touchElapsed * 1000;
671                     if (velocityX > enchant.widget._env.FLINGVEL || velocityY > enchant.widget._env.FLINGVEL) {
672                         var evt = new enchant.Event(enchant.Event.FLING);
673                         evt.x = this._startX;
674                         evt.y = this._startY;
675                         evt.ex = this._lastX;
676                         evt.ey = this._lastY;
677                         evt.velocityX = velocityX;
678                         evt.velocityY = velocityY;
679                         this._target.dispatchEvent(evt);
680                     }
681                     this._state = NOTOUCH;
682                     break;
683                 case HOLD:
684                     var evt = new enchant.Event(enchant.Event.RELEASE);
685                     evt.x = this._lastX;
686                     evt.y = this._lastY;
687                     this._target.dispatchEvent(evt);
688                     this._state = NOTOUCH;
689                     break;
690                 case NOMOVEDBL:
691                     var evt = new enchant.Event(enchant.Event.DOUBLETAP);
692                     evt.x = this._lastX;
693                     evt.y = this._lastY;
694                     this._target.dispatchEvent(evt);
695                     this._state = NOTOUCH;
696                     this._releaseElapsed = 0;
697                     break;
698                 case NOMOVE:
699                     this._state = WAITDBL;
700                     break;
701                 default:
702                     this._state = NOTOUCH;
703                     break;
704             }
705             this._touchElapsed = 0;
706             this._startX = 0;
707             this._startY = 0;
708         },
709         onenterframe: function(e) {
710             var elapsed = e.elapsed;
711             switch (this._state) {
712                 case WAITDBL:
713                     this._releaseElapsed += elapsed;
714                     if (this._releaseElapsed >= enchant.widget._env.DBLLIMIT) {
715                         var evt = new enchant.Event(enchant.Event.TAP);
716                         evt.x = this._lastX;
717                         evt.y = this._lastY;
718                         this._lastX = 0;
719                         this._lastY = 0;
