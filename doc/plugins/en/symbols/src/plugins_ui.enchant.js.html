 47 /**
 48  * 方向キーパッドのクラス: Pad
 49  * @scope enchant.ui.Pad
 50  */
 51 enchant.ui.Pad = enchant.Class.create(enchant.Sprite, {
 52     /**
 53      * 方向キーパッドオブジェクトを作成する。
 54      * @constructs
 55      * @extends enchant.Sprite
 56      */
 57     initialize: function() {
 58         var core = enchant.Core.instance;
 59         var image = core.assets['pad.png'];
 60         enchant.Sprite.call(this, image.width / 2, image.height);
 61         this.image = image;
 62         this.input = { left: false, right: false, up: false, down: false };
 63         this.addEventListener('touchstart', function(e) {
 64             this._updateInput(this._detectInput(e.localX, e.localY));
 65         });
 66         this.addEventListener('touchmove', function(e) {
 67             this._updateInput(this._detectInput(e.localX, e.localY));
 68         });
 69         this.addEventListener('touchend', function(e) {
 70             this._updateInput({ left: false, right: false, up: false, down: false });
 71         });
 72     },
 73     _detectInput: function(x, y) {
 74         x -= this.width / 2;
 75         y -= this.height / 2;
 76         var input = { left: false, right: false, up: false, down: false };
 77         if (x * x + y * y > 200) {
 78             if (x < 0 && y < x * x * 0.1 && y > x * x * -0.1) {
 79                 input.left = true;
 80             }
 81             if (x > 0 && y < x * x * 0.1 && y > x * x * -0.1) {
 82                 input.right = true;
 83             }
 84             if (y < 0 && x < y * y * 0.1 && x > y * y * -0.1) {
 85                 input.up = true;
 86             }
 87             if (y > 0 && x < y * y * 0.1 && x > y * y * -0.1) {
 88                 input.down = true;
 89             }
 90         }
 91         return input;
 92     },
 93     _updateInput: function(input) {
 94         var core = enchant.Core.instance;
 95         ['left', 'right', 'up', 'down'].forEach(function(type) {
 96             if (this.input[type] && !input[type]) {
 97                 core.dispatchEvent(new enchant.Event(type + 'buttonup'));
 98             }
 99             if (!this.input[type] && input[type]) {
100                 core.dispatchEvent(new enchant.Event(type + 'buttondown'));
101             }
102         }, this);
103         this.input = input;
104     }
105 });

107 /**
108  * アナログパッドのクラス: APad
109  * @scope enchant.ui.APad
110  */
111 enchant.ui.APad = enchant.Class.create(enchant.Group, {
112     /**
113      * アナログパッドオブジェクトを作成する。
114      * @constructs
115      * @extends enchant.Group
116      * @param mode
117      *   'direct': 入力ベクトルは正規化されない (大きさは 0~1 の間)
118      *   'normal': 入力ベクトルを常に正規化する (大きさは常に1となる)
119      */
120     initialize: function(mode) {
121         var core = enchant.Core.instance;
122         var image = core.assets['apad.png'];
123         var w = this.width = image.width;
124         var h = this.height = image.height;
125         enchant.Group.call(this);
126 
127         this.outside = new enchant.Sprite(w, h);
128         var outsideImage = new enchant.Surface(w, h);
129         outsideImage.draw(image, 0, 0, w, h / 4, 0, 0, w, h / 4);
130         outsideImage.draw(image, 0, h / 4 * 3, w, h / 4, 0, h / 4 * 3, w, h / 4);
131         outsideImage.draw(image, 0, h / 4, w / 4, h / 2, 0, h / 4, w / 4, h / 2);
132         outsideImage.draw(image, w / 4 * 3, h / 4, w / 4, h / 2, w / 4 * 3, h / 4, w / 4, h / 2);
133         this.outside.image = outsideImage;
134         this.inside = new enchant.Sprite(w / 2, h / 2);
135         var insideImage = new enchant.Surface(w / 2, h / 2);
136         insideImage.draw(image, w / 4, h / 4, w / 2, h / 2, 0, 0, w / 2, h / 2);
137         this.inside.image = insideImage;
138         this.r = w / 2;
139 
140         /**
141          * isTouched
142          * @type {Boolean}
143          * タッチされているかどうか
144          */
145         this.isTouched = false;
146 
147         /**
148          * vx, vy
149          * @type {Number}
150          * 入力ベクトルの(x, y)方向の大きさ
151          */
152         this.vx = 0;
153         this.vy = 0;
154 
155         /**
156          * rad, dist
157          * @type {Number}
158          * 入力ベクトルの極座標表示
159          * radは角度、distはベクトルの大きさを示す
160          */
161         this.rad = 0;
162         this.dist = 0;
163 
164         if (mode === 'direct') {
165             this.mode = 'direct';
166         } else {
167             this.mode = 'normal';
168         }
169         this._updateImage();
170         this.addChild(this.inside);
171         this.addChild(this.outside);
172         this.addEventListener('touchstart', function(e) {
173             this._detectInput(e.localX, e.localY);
174             this._calcPolar(e.localX, e.localY);
175             this._updateImage(e.localX, e.localY);
176             this._dispatchPadEvent('apadstart');
177             this.isTouched = true;
178         });
179         this.addEventListener('touchmove', function(e) {
180             this._detectInput(e.localX, e.localY);
181             this._calcPolar(e.localX, e.localY);
182             this._updateImage(e.localX, e.localY);
183             this._dispatchPadEvent('apadmove');
184         });
185         this.addEventListener('touchend', function(e) {
186             this._detectInput(this.width / 2, this.height / 2);
187             this._calcPolar(this.width / 2, this.height / 2);
188             this._updateImage(this.width / 2, this.height / 2);
189             this._dispatchPadEvent('apadend');
190             this.isTouched = false;
191         });
192     },
193     _dispatchPadEvent: function(type) {
194         var e = new enchant.Event(type);
195         e.vx = this.vx;
196         e.vy = this.vy;
197         e.rad = this.rad;
198         e.dist = this.dist;
199         this.dispatchEvent(e);
200     },
201     _updateImage: function(x, y) {
202         x -= this.width / 2;
203         y -= this.height / 2;
204         this.inside.x = this.vx * (this.r - 10) + 25;
205         this.inside.y = this.vy * (this.r - 10) + 25;
206     },
207     _detectInput: function(x, y) {
208         x -= this.width / 2;
209         y -= this.height / 2;
210         var distance = Math.sqrt(x * x + y * y);
211         var tan = y / x;
212         var rad = Math.atan(tan);
213         var dir = x / Math.abs(x);
214         if (distance === 0) {
215             this.vx = 0;
216             this.vy = 0;
217         } else if (x === 0) {
218             this.vx = 0;
219             if (this.mode === 'direct') {
220                 this.vy = (y / this.r);
221             } else {
222                 dir = y / Math.abs(y);
223                 this.vy = Math.pow((y / this.r), 2) * dir;
224             }
225         } else if (distance < this.r) {
226             if (this.mode === 'direct') {
227                 this.vx = (x / this.r);
228                 this.vy = (y / this.r);
229             } else {
230                 this.vx = Math.pow((distance / this.r), 2) * Math.cos(rad) * dir;
231                 this.vy = Math.pow((distance / this.r), 2) * Math.sin(rad) * dir;
232             }
233         } else {
234             this.vx = Math.cos(rad) * dir;
235             this.vy = Math.sin(rad) * dir;
236         }
237     },
238     _calcPolar: function(x, y) {
239         x -= this.width / 2;
240         y -= this.height / 2;
241         var add = 0;
242         var rad = 0;
243         var dist = Math.sqrt(x * x + y * y);
244         if (dist > this.r) {
245             dist = this.r;
246         }
247         dist /= this.r;
248         if (this.mode === 'normal') {
249             dist *= dist;
250         }
251         if (x >= 0 && y < 0) {
252             add = Math.PI / 2 * 3;
253             rad = x / y;
254         } else if (x < 0 && y <= 0) {
255             add = Math.PI;
256             rad = y / x;
257         } else if (x <= 0 && y > 0) {
258             add = Math.PI / 2;
259             rad = x / y;
260         } else if (x > 0 && y >= 0) {
261             add = 0;
262             rad = y / x;
263         }
264         if (x === 0 || y === 0) {
265             rad = 0;
266         }
267         this.rad = Math.abs(Math.atan(rad)) + add;
268         this.dist = dist;
269     }
270 });

272 /**
273  * ボタンオブジェクトのクラス: Button
274  * available in only DOMGroup
275  *
276  * @scope enchant.ui.Button.prototype
277  */
278 enchant.ui.Button = enchant.Class.create(enchant.Entity, {
279     /**
280      * ボタンオブジェクトを作成する。
281      * @constructs
282      * @extends enchant.Entity
283      */
284     initialize: function(text, theme, height, width) {
285         enchant.Entity.call(this);
286 
287         // ベンタプレフィクスを判定
288         this.VENDOR_PREFIX = (function() {
289             var ua = navigator.userAgent;
290             if (ua.indexOf('Opera') !== -1) {
291                 return '-O-';
292             } else if (ua.indexOf('MSIE') !== -1) {
293                 return '-ms-';
294             } else if (ua.indexOf('WebKit') !== -1) {
295                 return '-webkit-';
296             } else if (navigator.product === 'Gecko') {
297                 return '-Moz-';
298             } else {
299                 return '';
300             }
301         }());
302 
303         this.width = width || null;
304         this.height = height || null;
305         this.text = text;
306         this.pressed = false;
307 
308         // デフォルトのスタイル (テーマで上書き可能)
309         var style = this._element.style;
310         style["display"] = "inline-block";
311         style["font-size"] = "12px";
312         style["height"] = "2em";
313         style["line-height"] = "2em";
314         style["min-width"] = "2em";
315         style["padding"] = "2px 10px";
316         style["text-align"] = "center";
317         style["font-weight"] = "bold";
318         style[this.VENDOR_PREFIX + "border-radius"] = "0.5em";
319 
320         // テーマの指定がなければ "dark" を使う
321         theme = theme || "dark";
322 
323         if (typeof theme === "string") {
324             // theme 引数が string なら、その名前のデフォルトテーマを使う
325             this.theme = enchant.ui.Button.DEFAULT_THEME[theme];
326         } else {
327             // theme 引数が object なら、その引数をテーマとして扱う
328             this.theme = theme;
329         }
330 
331         // テーマを適用する
332         this._applyTheme(this.theme.normal);
333 
334         // タッチしたときの挙動
335         this.addEventListener("touchstart", function() {
336             this._applyTheme(this.theme.active);
337             this.pressed = true;
338             this.y++;
339         });
340 
341         // タッチが離されたときの挙動
342         this.addEventListener("touchend", function() {
343             this._applyTheme(this.theme.normal);
344             this.pressed = false;
345             this.y--;
346         });
347     },
348     _applyTheme: function(theme) {
349         var style = this._element.style;
350         for (var i in theme) {
351             if (theme.hasOwnProperty(i)) {
352                 var prop = i.replace(/\-VENDOR\-/, this.VENDOR_PREFIX);
353                 style[prop] = (theme[i] + '').replace(/\-VENDOR\-/, this.VENDOR_PREFIX);
354             }
355         }
356     },
357     /**
358      * 表示するテキスト
359      * @type {String}
360      */
361     text: {
362         get: function() {
363             return this._element.innerHTML;
364         },
365         set: function(text) {
366             this._element.innerHTML = text;
367         }
368     },
369     /**
370      * フォントサイズ
371      */
372     size: {
373         get: function() {
374             return this._element.style["fontSize"];
375         },
376         set: function(size) {
377             this._element.style["fontSize"] = size;
378         }
379     },
380     /**
381      * フォントの指定
382      * @type {String}
383      */
384     font: {
385         get: function() {
386             return this._style.font;
387         },
388         set: function(font) {
389             this._style.font = font;
390         }
391     },
392     /**
393      * Text color settings.
394      * CSS 'color' can be set to same format as properties.
395      * @type {String}
396      */
397     color: {
398         get: function() {
399             return this._style.color;
400         },
401         set: function(color) {
402             this._style.color = color;
403         }
404     }
405 });
